  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const check = await client.query(`SELECT 1 FROM schema_migrations WHERE id = 'bridge_v6_agent_messages'`);
    if (check.rowCount && check.rowCount > 0) { await client.query('COMMIT'); return; }
    // ... DDL here ...
    await client.query(`INSERT INTO schema_migrations (id) VALUES ('bridge_v6_agent_messages')`);
    await client.query('COMMIT');
  } catch (err) { await client.query('ROLLBACK'); throw err; }
  finally { client.release(); }
}
