2. Add `logWatcherFinding` function (not exported — internal helper):
   ```typescript
   async function logWatcherFinding(
     finding: { id: string; title: string; summary: string; importance: string; source_type: string },
     watcher: { id: string; name: string; project_id: string; notify_email: string | null },
     jobId: string,
   ): Promise<void> {
     // 1. Insert into agent_activity for project feed
     const sourceBadge = `[${finding.source_type}]`;
     const activitySummary = `${sourceBadge} ${finding.title}`;
     const activityDetail = JSON.stringify({
       finding_id: finding.id,
       watcher_id: watcher.id,
       watcher_name: watcher.name,
       summary: finding.summary,
       source_type: finding.source_type,
       importance: finding.importance,
     });
     
     await pool.query(
       `INSERT INTO agent_activity (agent_id, job_id, project_id, event_type, summary, detail)
        VALUES ('system', $1, $2, $3, $4, $5)`,
       [jobId, watcher.project_id, `watcher_finding_${finding.source_type}`, activitySummary, activityDetail]
     );
     
     // 2. SSE push for real-time feed updates
     const { emitSSE } = await import('./scheduler.js');
     await emitSSE('project:activity', {
       agent_id: 'system',
       job_id: jobId,
       project_id: watcher.project_id,
       event_type: `watcher_finding_${finding.source_type}`,
       summary: activitySummary,
       detail: activityDetail,
       created_at: Date.now() / 1000,
     }).catch(() => {});
     
     // 3. Notification for important/critical findings
     if (finding.importance === 'important' || finding.importance === 'critical') {
       // In-app notification via SSE
       await emitSSE('watcher:important-finding', {
         finding_id: finding.id,
         watcher_id: watcher.id,
         watcher_name: watcher.name,
         project_id: watcher.project_id,
         title: finding.title,
         summary: finding.summary,
         source_type: finding.source_type,
         importance: finding.importance,
       }).catch(() => {});
       
       // Optional email notification
       if (watcher.notify_email) {
         try {
           const { sendEmail } = await import('./email.js');
           await sendEmail({
             to: watcher.notify_email,
             subject: `[Porter] ${finding.importance === 'critical' ? 'CRITICAL' : 'Important'}: ${finding.title}`,
             body: `<h3>${finding.title}</h3>
                    <p>${finding.summary}</p>
                    <p><strong>Watcher:</strong> ${watcher.name}</p>
                    <p><strong>Source:</strong> ${finding.source_type}</p>
                    <p><em>View in Porter for full details.</em></p>`,
           });
         } catch (emailErr) {
           console.error('[watcher] email notification failed:', emailErr instanceof Error ? emailErr.message : emailErr);
           // Email failure is non-blocking — finding is already stored
         }
       }
     }
   }
   ```

   Endpoints:

   a) `POST /` — Create a watcher (PMN-01)
      - Body: `{ project_id, name, watcher_type, schedule_interval_sec?, config, notify_email? }`
      - Validate: watcher_type must be one of 'web_search', 'rss_feed', 'email_monitor', 'custom'
      - Validate: project_id exists in projects table
      - Generate UUID for id
      - Compute schedule_cron from schedule_interval_sec (simple mapping: 3600=hourly, 21600=every 6h, 86400=daily; else store as '*/X * * * *')
      - Set next_run_at = NOW() + schedule_interval_sec
      - INSERT INTO project_watchers with all fields
      - Return ok({ watcher: inserted row })

   b) `GET /` — List all watchers (for PMN-05 ops view)
      - Query params: project_id (optional filter), status (optional filter), limit (default 50), offset (default 0)
      - SELECT pw.*, p.name AS project_name FROM project_watchers pw LEFT JOIN projects p ON p.id = pw.project_id
      - Dynamic WHERE based on filters
      - ORDER BY pw.created_at DESC
      - Return ok({ watchers: rows, total })

   c) `GET /:watcherId` — Single watcher detail
      - SELECT pw.*, p.name AS project_name FROM project_watchers pw LEFT JOIN projects p ON p.id = pw.project_id WHERE pw.id = $1
      - 404 if not found
      - Return ok(row)

   d) `PATCH /:watcherId` — Update watcher config
      - Body: partial update of { name?, schedule_interval_sec?, config?, status?, notify_email? }
      - Build dynamic SET clause (only set provided fields)
      - Always SET updated_at = NOW()
      - If schedule_interval_sec changed, recalculate next_run_at
      - Return ok({ watcher: updated row })

   e) `DELETE /:watcherId` — Remove a watcher
      - DELETE FROM watcher_findings WHERE watcher_id = $1 (cascade findings)
      - DELETE FROM project_watchers WHERE id = $1
      - Return ok({ deleted: true })

   f) `GET /:watcherId/findings` — List findings for a watcher
      - Query params: importance (optional filter), limit (default 50), offset (default 0)
      - SELECT * FROM watcher_findings WHERE watcher_id = $1 ORDER BY created_at DESC
      - Return ok({ findings: rows, total })

   g) `POST /:watcherId/run` — Trigger immediate watcher run
      - Check watcher exists and is active
      - Create agent_job with trigger_type='watcher_run', trigger_data={watcher_id}, source='watcher', scheduled_for=NOW()
      - Return ok({ job_id })
