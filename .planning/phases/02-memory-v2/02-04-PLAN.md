2. **Rebuild loadMemory() JS** — replace the entire function with:

```javascript
function loadMemory(agentId) {
  var feed = document.getElementById('recall-feed');
  var empty = document.getElementById('recall-feed-empty');
  if (!feed) return;
  feed.innerHTML = '<div style="padding:16px;color:var(--text3);font-size:12px">Loading...</div>';

  var url = '/api/memory/feed' + (agentId ? '?agent_id=' + encodeURIComponent(agentId) : '');
  fetch(url, {credentials:'include'}).then(function(r){return r.json()}).then(function(data){
    feed.innerHTML = '';
    if (!data.items || data.items.length === 0) {
      if (empty) empty.style.display = '';
      return;
    }
    if (empty) empty.style.display = 'none';
    data.items.forEach(function(item) { _recallFeedAppend(feed, item); });

    // Update badge count
    var badge = document.getElementById('recall-badge');
    if (badge) { badge.textContent = '0'; badge.style.display = 'none'; }

    // Mark as read
    fetch('/api/memory/mark-read', {method:'POST', credentials:'include'});
  }).catch(function(e){ feed.innerHTML = '<div style="padding:16px;color:var(--danger)">Failed to load: ' + e.message + '</div>'; });
}

function _recallFeedPrepend(data) {
  var feed = document.getElementById('recall-feed');
  var empty = document.getElementById('recall-feed-empty');
  if (!feed) return;
  if (empty) empty.style.display = 'none';

  // Apply scope filter
  var filter = document.getElementById('recall-scope-filter');
  if (filter && filter.value !== 'all' && data.scope !== filter.value) return;

  var row = _recallFeedRow(data);
  feed.insertBefore(row, feed.firstChild);

  // Update badge
  var badge = document.getElementById('recall-badge');
  if (badge) {
    var count = parseInt(badge.textContent || '0') + 1;
    badge.textContent = count;
    badge.style.display = '';
  }
}

function _recallFeedAppend(container, data) {
  container.appendChild(_recallFeedRow(data));
}

function _recallFeedRow(data) {
  var row = document.createElement('div');
  row.className = 'recall-row';
  row.style.cssText = 'display:flex;align-items:center;gap:8px;padding:6px 0;border-bottom:1px solid var(--border);font-size:12px;cursor:default';

  var icon = data.action === 'learned' ? '&#128161;' : data.action === 'promoted' ? '&#11088;' : data.action === 'dismissed' ? '&#128683;' : '&#128260;';
  var scopeColor = data.scope === 'global' ? 'var(--accent)' : data.scope === 'project' ? 'var(--success, #4ade80)' : 'var(--warning, #fbbf24)';

  row.innerHTML = '<span style="flex-shrink:0;width:18px;text-align:center">' + icon + '</span>'
    + '<span style="flex:1;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;color:var(--text)">' + (typeof escHtml === 'function' ? escHtml(data.text || '') : (data.text || '').replace(/</g,'&lt;')) + '</span>'
    + '<span style="flex-shrink:0;font-size:10px;padding:1px 6px;border-radius:3px;background:' + scopeColor + ';color:#000;opacity:0.8">' + (data.scope || 'global') + '</span>'
    + '<span style="flex-shrink:0;font-size:10px;color:var(--text3)">' + _recallTimeAgo(data.ts) + '</span>';
  return row;
}

function _recallTimeAgo(ts) {
  if (!ts) return '';
  var diff = (Date.now()/1000) - ts;
  if (diff < 60) return 'now';
  if (diff < 3600) return Math.floor(diff/60) + 'm';
  if (diff < 86400) return Math.floor(diff/3600) + 'h';
  return Math.floor(diff/86400) + 'd';
}
```
