def _count_recent_signals(since_ts):
    """Count signals inserted after a given timestamp."""
    conn = sqlite3.connect(PORTER_DB)
    conn.row_factory = sqlite3.Row
    count = conn.execute(
        "SELECT COUNT(*) as cnt FROM memories WHERE memory_kind='signal' AND created_at > ?",
        (since_ts,)
    ).fetchone()['cnt']
    conn.close()
    return count

    req = urllib.request.Request(
        f"{PORTER_URL}{path}",
        data=json.dumps(data).encode(),
        headers={"Content-Type": "application/json"}
    )
    try:
        with urllib.request.urlopen(req, timeout=5) as resp:

    idx = src.index('def _mem_inject_for_dispatch')
    fn_body = src[idx:src.index('\ndef ', idx + 1)]
    assert '_project_is_private' in fn_body, "Privacy check missing from injection"

def test_scope_isolation_in_db():
    """Insert project-A memory, verify it does NOT appear in project-B query."""
    conn = sqlite3.connect(PORTER_DB)
    conn.row_factory = sqlite3.Row
    test_id = f"test_scope_{int(time.time())}"

    # Insert a project-scoped memory for project-A
    conn.execute(
        "INSERT INTO memories (memory_kind, text, scope, scope_id, trust_tier, status, source_type, created_at) "
        "VALUES ('concept', ?, 'project', 'project-A-test', 'medium', 'active', 'test', ?)",
        (f"Test scope isolation {test_id}", time.time())
    )
    conn.commit()

    # Query for project-B -- must NOT find the project-A memory
    rows = conn.execute(
        "SELECT * FROM memories WHERE scope='project' AND scope_id='project-B-test' AND text LIKE ?",
        (f"%{test_id}%",)
    ).fetchall()
    assert len(rows) == 0, f"Project-A memory leaked to project-B: found {len(rows)} rows"

    # Cleanup
    conn.execute("DELETE FROM memories WHERE text LIKE ?", (f"%{test_id}%",))
    conn.commit()
    conn.close()

def test_tiered_injection_labels():
    """Injection must use [DIRECTIVE], [CONCEPT], [EPISODE] tier labels."""
