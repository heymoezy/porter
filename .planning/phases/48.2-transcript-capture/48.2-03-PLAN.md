```javascript
#!/usr/bin/env node
// Phase 48.2 -- Porter Stop hook (TRC-03 + TRC-08)
// Captures assistant turns from the JSONL transcript_path using a per-session
// byte-offset bookmark. NEVER writes to stdout (decision:block would force
// Claude to continue per Risk 2). Idempotent via UNIQUE(session_id, turn_index)
// + ON CONFLICT DO NOTHING at the backend.

const fs = require('node:fs');
const path = require('node:path');
const http = require('node:http');
const readline = require('node:readline');

const BOOKMARK_DIR = '/tmp/porter-transcript-bookmark';
const PORTER_HOST = '127.0.0.1';
const PORTER_PORT = 3001;
const CAPTURE_PATH = '/api/v1/intellect/transcript/turn';
const FLUSH_DELAY_MS = 250;
const MAX_CONTENT = 32 * 1024;

function readStdinJson(timeoutMs = 1000) {
  return new Promise((resolve) => {
    let buf = '';
    const t = setTimeout(() => resolve(null), timeoutMs);
    process.stdin.setEncoding('utf8');
    process.stdin.on('data', (c) => { buf += c; });
    process.stdin.on('end', () => {
      clearTimeout(t);
      try { resolve(JSON.parse(buf || '{}')); } catch { resolve(null); }
    });
    process.stdin.on('error', () => { clearTimeout(t); resolve(null); });
  });
}

function postTurn(payload) {
  return new Promise((resolve) => {
    const body = JSON.stringify(payload);
    const req = http.request({
      hostname: PORTER_HOST,
      port: PORTER_PORT,
      path: CAPTURE_PATH,
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(body),
      },
      timeout: 2000,
    }, (res) => {
      // Drain to allow connection reuse / avoid leaks
      res.on('data', () => {});
      res.on('end', () => resolve());
    });
    req.on('error', () => resolve());
    req.on('timeout', () => { try { req.destroy(); } catch {} resolve(); });
    req.write(body);
    req.end();
  });
}

async function main() {
  const data = await readStdinJson();
  if (!data) process.exit(0);

  const sessionId = data.session_id;
  const transcriptPath = data.transcript_path;
  const cwd = data.cwd || null;

  if (!sessionId || !transcriptPath) process.exit(0);
  if (!fs.existsSync(transcriptPath)) process.exit(0);

  // Flush-race breathing room (Pitfall 2; Anthropic Issue #8564)
  await new Promise((r) => setTimeout(r, FLUSH_DELAY_MS));

  try { fs.mkdirSync(BOOKMARK_DIR, { recursive: true }); } catch {}
  const bookmarkPath = path.join(BOOKMARK_DIR, `${sessionId}.offset`);
  let offset = 0;
  try { offset = parseInt(fs.readFileSync(bookmarkPath, 'utf8'), 10) || 0; } catch {}

  const stat = fs.statSync(transcriptPath);
  if (stat.size <= offset) process.exit(0);

  const stream = fs.createReadStream(transcriptPath, { start: offset });
  const rl = readline.createInterface({ input: stream });

  try {
    for await (const line of rl) {
      if (!line.trim()) continue;
      let row;
      try { row = JSON.parse(line); } catch { continue; }
      if (!row || row.type !== 'assistant') continue;
      if (row.isSidechain) continue;
      const msgRole = row.message && row.message.role;
      if (msgRole !== 'assistant') continue;
      const blocks = (row.message && row.message.content) || [];
      const text = blocks
        .filter((b) => b && b.type === 'text' && typeof b.text === 'string')
        .map((b) => b.text)
        .join('\n')
        .trim();
      if (!text) continue;
      await postTurn({
        session_id: sessionId,
        cwd,
        role: 'assistant',
        content: text.slice(0, MAX_CONTENT),
        captured_at: row.timestamp || new Date().toISOString(),
      });
    }
    // Advance bookmark only after a successful full parse pass.
    try { fs.writeFileSync(bookmarkPath, String(stat.size)); } catch {}
  } catch (e) {
    try { process.stderr.write(`[porter-stop] parse error: ${e && e.message ? e.message : e}\n`); } catch {}
  }

  process.exit(0);
}

main().catch(() => process.exit(0));
```
