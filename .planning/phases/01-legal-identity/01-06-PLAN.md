```typescript
// GET /api/auth/singpass
// Initiates SingPass PAR flow — returns redirect URL for browser
// Mobile app calls this, gets a URL, opens it in the system browser
import { NextRequest, NextResponse } from 'next/server'
import { cookies } from 'next/headers'
import { randomUUID } from 'crypto'
import { initiateSingPassAuth } from '@/services/identity/singpass'

export async function GET(request: NextRequest) {
  try {
    // Session ID ties the OAuth state to this browser session
    const sessionId = randomUUID()
    const redirectUrl = await initiateSingPassAuth(sessionId)

    // Store session ID in cookie for callback validation
    const cookieStore = await cookies()
    cookieStore.set('singpass_session_id', sessionId, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production',
      sameSite: 'lax',
      maxAge: 600, // 10 minutes — well beyond the 60s request_uri expiry
    })

    return NextResponse.redirect(redirectUrl)
  } catch (error) {
    console.error('[singpass/route] Failed to initiate SingPass auth:', error)
    return NextResponse.redirect(
      new URL('/worker/singpass?error=initiation_failed', request.url)
    )
  }
}
```

```typescript
// GET /api/auth/singpass/callback
// SingPass redirects here after user consents — exchanges code for tokens, fetches person data
import { NextRequest, NextResponse } from 'next/server'
import { cookies } from 'next/headers'
import { getOAuthSession, clearOAuthSession } from '@/services/identity/singpass'
import { parseMyInfoPerson } from '@/services/identity/myinfo-parser'
import { db } from '@/db'
import { workerProfiles } from '@/db/schema/workers'
import { workerIdentityVerifications } from '@/db/schema/verifications'
import { eq } from 'drizzle-orm'
import { createServerClient } from '@supabase/ssr'

export async function GET(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams
  const code = searchParams.get('code')
  const state = searchParams.get('state')
  const error = searchParams.get('error')

  // SingPass returned an error (e.g., user cancelled)
  if (error) {
    return NextResponse.redirect(
      new URL(`/worker/singpass?error=${error}`, request.url)
    )
  }

  if (!code || !state) {
    return NextResponse.redirect(
      new URL('/worker/singpass?error=missing_params', request.url)
    )
  }

  const cookieStore = await cookies()
  const sessionId = cookieStore.get('singpass_session_id')?.value

  if (!sessionId) {
    return NextResponse.redirect(
      new URL('/worker/singpass?error=session_expired', request.url)
    )
  }

  const session = getOAuthSession(sessionId)
  if (!session) {
    return NextResponse.redirect(
      new URL('/worker/singpass?error=session_not_found', request.url)
    )
  }

  // Validate state matches — prevents CSRF
  if (session.state !== state) {
    clearOAuthSession(sessionId)
    return NextResponse.redirect(
      new URL('/worker/singpass?error=state_mismatch', request.url)
    )
  }

  try {
    // TODO: Exchange code for tokens and fetch person data via myInfoClient
    // The exact token exchange depends on the myinfo-connector-v4-nodejs API
    // For now, log the successful callback and update onboarding step
    // Full token exchange implementation wires myInfoClient.getPersonData()
    // Reference: github.com/singpass/myinfo-connector-v4-nodejs README

    // Get the authenticated Supabase user from the current session
    // Worker must already have a Supabase auth account (created at account creation step before SingPass)
    const supabase = createServerClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_ROLE_KEY!,
      { cookies: { getAll: () => cookieStore.getAll() } }
    )
    const { data: { user } } = await supabase.auth.getUser()

    if (!user) {
      clearOAuthSession(sessionId)
      return NextResponse.redirect(new URL('/worker/signup?error=not_authenticated', request.url))
    }

    // Update worker onboarding step to singpass_complete
    await db
      .update(workerProfiles)
      .set({ onboardingStep: 'singpass_complete', updatedAt: new Date() })
      .where(eq(workerProfiles.supabaseUserId, user.id))

    clearOAuthSession(sessionId)
    cookieStore.delete('singpass_session_id')

    // Redirect to next onboarding step
    return NextResponse.redirect(new URL('/worker/profile', request.url))
  } catch (error) {
    console.error('[singpass/callback] Token exchange failed:', error)
    clearOAuthSession(sessionId)
    return NextResponse.redirect(
      new URL('/worker/singpass?error=callback_failed', request.url)
    )
  }
}
```

```tsx
// Step 4: Food hygiene cert upload (F&B workers only)
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'

export default function WorkerCertUploadPage() {
  const router = useRouter()
  const [file, setFile] = useState<File | null>(null)
  const [expiryDate, setExpiryDate] = useState('')
  const [isUploading, setIsUploading] = useState(false)
  const [uploadError, setUploadError] = useState<string | null>(null)

  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const selected = e.target.files?.[0]
    if (!selected) return
    const allowedTypes = ['image/jpeg', 'image/png', 'image/webp', 'application/pdf']
    if (!allowedTypes.includes(selected.type)) {
      setUploadError('Please upload an image (JPG, PNG, WebP) or PDF.')
      return
    }
    if (selected.size > 10 * 1024 * 1024) { // 10MB limit
      setUploadError('File must be under 10MB.')
      return
    }
    setUploadError(null)
    setFile(selected)
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!file || !expiryDate) return
    setIsUploading(true)

    const formData = new FormData()
    formData.append('cert', file)
    formData.append('expiryDate', expiryDate)
    formData.append('step', 'cert_upload')

    const res = await fetch('/api/auth/worker-onboarding', {
      method: 'POST',
      body: formData,
    })

    if (res.ok) {
      router.push('/worker/contract')
    } else {
      setUploadError('Upload failed. Please try again.')
      setIsUploading(false)
    }
  }

  return (
    <div className="flex min-h-screen flex-col items-center justify-center px-4">
      <div className="w-full max-w-md space-y-6">
        <div>
          <h1 className="text-2xl font-bold text-gray-900">Food hygiene certificate</h1>
          <p className="mt-2 text-gray-600">
            Upload a photo or scan of your WSQ Food Safety certificate.
            Our team will verify it within 24 hours.
          </p>
        </div>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-gray-700">Certificate image</label>
            <input
              type="file"
              accept="image/jpeg,image/png,image/webp,application/pdf"
              onChange={handleFileChange}
              required
              className="mt-1 block w-full text-sm text-gray-500"
            />
            {file && (
              <p className="mt-1 text-sm text-green-600">Selected: {file.name}</p>
            )}
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700">Certificate expiry date</label>
            <input
              type="date"
              value={expiryDate}
              onChange={e => setExpiryDate(e.target.value)}
              required
              min={new Date().toISOString().split('T')[0]}
              className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2"
            />
          </div>
          {uploadError && (
            <p className="text-sm text-red-600">{uploadError}</p>
          )}
          <button
            type="submit"
            disabled={!file || !expiryDate || isUploading}
            className="w-full rounded-lg bg-blue-600 px-6 py-3 text-white font-medium
                       hover:bg-blue-700 disabled:opacity-50"
          >
            {isUploading ? 'Uploading…' : 'Upload and continue'}
          </button>
        </form>
        <p className="text-center text-xs text-gray-500">
          While your cert is being reviewed, you can work retail shifts immediately.
        </p>
      </div>
    </div>
  )
}
```

```typescript
// POST /api/auth/worker-onboarding
// Progresses worker through onboarding steps — validates step order, updates DB
import { NextRequest, NextResponse } from 'next/server'
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import { db } from '@/db'
import { workerProfiles } from '@/db/schema/workers'
import { eq } from 'drizzle-orm'
import { recordWorkerContractAcceptance } from '@/services/identity/contract'

const CURRENT_CONTRACT_VERSION = process.env.CURRENT_CONTRACT_VERSION ?? '1.0'

export async function POST(request: NextRequest) {
  const cookieStore = await cookies()
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { cookies: { getAll: () => cookieStore.getAll() } }
  )

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const contentType = request.headers.get('content-type') ?? ''
  let body: Record<string, unknown>

  if (contentType.includes('multipart/form-data')) {
    const formData = await request.formData()
    body = Object.fromEntries(formData.entries())
  } else {
    body = await request.json()
  }

  const { step } = body

  const worker = await db.query.workerProfiles.findFirst({
    where: eq(workerProfiles.supabaseUserId, user.id),
  })

  if (!worker) {
    return NextResponse.json({ error: 'Worker profile not found' }, { status: 404 })
  }

  switch (step) {
    case 'profile': {
      const { preferredName, phone } = body as { preferredName: string; phone: string }
      if (!preferredName || !phone) {
        return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
      }
      await db.update(workerProfiles).set({
        preferredName: String(preferredName),
        phoneNumber: String(phone),
        onboardingStep: 'profile_complete',
        updatedAt: new Date(),
      }).where(eq(workerProfiles.id, worker.id))
      return NextResponse.json({ ok: true, nextStep: 'industry' })
    }

    case 'industry': {
      const { industry } = body as { industry: string }
      if (!['fnb', 'retail', 'both'].includes(industry)) {
        return NextResponse.json({ error: 'Invalid industry' }, { status: 400 })
      }
      await db.update(workerProfiles).set({
        industry: industry as 'fnb' | 'retail' | 'both',
        onboardingStep: 'industry_selected',
        updatedAt: new Date(),
      }).where(eq(workerProfiles.id, worker.id))
      const nextStep = industry === 'retail' ? 'contract' : 'cert_upload'
      return NextResponse.json({ ok: true, nextStep })
    }

    case 'cert_upload': {
      // Cert upload handled by separate cert storage API (Phase 2 wires Supabase Storage)
      // Set onboarding_step to cert_pending — cert is awaiting admin review
      await db.update(workerProfiles).set({
        onboardingStep: 'cert_pending',
        updatedAt: new Date(),
      }).where(eq(workerProfiles.id, worker.id))
      return NextResponse.json({ ok: true, nextStep: 'contract' })
    }

    case 'contract': {
      const { accepted } = body as { accepted: boolean }
      if (!accepted) {
        return NextResponse.json({ error: 'Contract not accepted' }, { status: 400 })
      }
      const ipAddress = request.headers.get('x-forwarded-for')?.split(',')[0] ?? undefined
      const userAgent = request.headers.get('user-agent') ?? undefined
      await recordWorkerContractAcceptance(worker.id, CURRENT_CONTRACT_VERSION, ipAddress, userAgent)
      await db.update(workerProfiles).set({
        onboardingStep: 'contract_accepted',
        canBrowseShifts: true,
        updatedAt: new Date(),
      }).where(eq(workerProfiles.id, worker.id))
      return NextResponse.json({ ok: true, nextStep: 'complete' })
    }

    default:
      return NextResponse.json({ error: 'Unknown step' }, { status: 400 })
  }
}
```
