```typescript
import { generateKeyPair, exportJWK, exportSPKI } from 'jose'

// Module-level cache — key pair persists for server lifetime
// In production, persist to env var or secrets manager to survive restarts
let cachedKeyPair: { privateKey: CryptoKey; publicKey: CryptoKey; kid: string } | null = null

export async function getOrCreateKeyPair() {
  if (cachedKeyPair) return cachedKeyPair

  const { privateKey, publicKey } = await generateKeyPair('RS256', {
    modulusLength: 2048,
    extractable: true,
  })

  const kid = `shiftsg-${Date.now()}`
  cachedKeyPair = { privateKey, publicKey, kid }
  return cachedKeyPair
}

export async function exportPublicJWKS() {
  const { publicKey, kid } = await getOrCreateKeyPair()
  const jwk = await exportJWK(publicKey)
  return {
    keys: [
      {
        ...jwk,
        alg: 'RS256',
        use: 'sig',
        kid,
      },
    ],
  }
}

export async function getPrivateKey() {
  const { privateKey } = await getOrCreateKeyPair()
  return privateKey
}
```

```typescript
import { z } from 'zod'
import { createHash } from 'crypto'

// MyInfo v5 person data shape (fields used by ShiftSG)
// Full schema: developer.singpass.gov.sg/docs/myinfo-v5/person-data
export const MyInfoPersonSchema = z.object({
  name: z.object({ value: z.string() }),
  dob: z.object({ value: z.string() }), // YYYY-MM-DD
  residentialaddress: z.object({
    block: z.object({ value: z.string().optional() }).optional(),
    street: z.object({ value: z.string().optional() }).optional(),
    postal: z.object({ value: z.string().optional() }).optional(),
  }).optional(),
  nationality: z.object({ code: z.string() }), // e.g., 'SG', 'PR'
  passtype: z.object({ code: z.string().optional() }).optional(), // 'S_PASS', 'EP', 'WP'
  workpassstatus: z.object({ code: z.string().optional() }).optional(),
  // uinfin is the NRIC — present in the sub claim of the ID token, NOT in person payload
  // We receive it separately from the ID token claims
})

export type MyInfoPerson = z.infer<typeof MyInfoPersonSchema>

export interface ParsedIdentity {
  name: string
  dob: string
  nationality: string      // 'SINGAPOREAN' | 'PR' | 'FOREIGNER'
  passType: string | null  // 'S_PASS' | 'EP' | 'WP' | null
  postalCode: string | null
  nricHash: string         // SHA-256(NRIC + salt) — NEVER raw NRIC
}

export function parseMyInfoPerson(payload: unknown, nric: string): ParsedIdentity {
  const data = MyInfoPersonSchema.parse(payload)

  const nationality = resolveNationality(data.nationality.code, data.passtype?.code)

  return {
    name: data.name.value,
    dob: data.dob.value,
    nationality,
    passType: data.passtype?.code ?? null,
    postalCode: data.residentialaddress?.postal?.value ?? null,
    nricHash: hashNric(nric),
  }
}

export function hashNric(nric: string): string {
  const salt = process.env.NRIC_HASH_SALT
  if (!salt) throw new Error('NRIC_HASH_SALT env var must be set before hashing')
  return createHash('sha256').update(nric + salt).digest('hex')
}

function resolveNationality(nationalityCode: string, passTypeCode?: string): string {
  if (nationalityCode === 'SG') return 'SINGAPOREAN'
  if (nationalityCode === 'PR') return 'PR'
  return 'FOREIGNER'
}
```

```typescript
import { randomUUID } from 'crypto'
import { SignJWT, generateKeyPair as joseGenerateKeyPair, exportJWK } from 'jose'
import { getPrivateKey, getOrCreateKeyPair } from '@/integrations/singpass/jwks'

// In-memory session store — replace with Redis or Supabase in production
const oauthSessions = new Map<string, OAuthSession>()

interface OAuthSession {
  codeVerifier: string
  state: string
  nonce: string
  dpopPrivateKey: CryptoKey
  dpopPublicJWK: object
  createdAt: number
}

export async function generatePKCEPair(): Promise<{ codeVerifier: string; codeChallenge: string }> {
  const array = new Uint8Array(32)
  crypto.getRandomValues(array)
  const codeVerifier = Buffer.from(array).toString('base64url')

  const encoder = new TextEncoder()
  const data = encoder.encode(codeVerifier)
  const digest = await crypto.subtle.digest('SHA-256', data)
  const codeChallenge = Buffer.from(digest).toString('base64url')

  return { codeVerifier, codeChallenge }
}

export async function generateDpopKeyPair() {
  const { privateKey, publicKey } = await joseGenerateKeyPair('ES256', { extractable: true })
  const publicJWK = await exportJWK(publicKey)
  return { privateKey, publicJWK }
}

export async function initiateSingPassAuth(sessionId: string): Promise<string> {
  const { codeVerifier, codeChallenge } = await generatePKCEPair()
  const { privateKey: dpopPrivateKey, publicJWK: dpopPublicJWK } = await generateDpopKeyPair()
  const state = randomUUID() // 36 chars — satisfies ≥30 requirement
  const nonce = randomUUID()

  const parEndpoint = process.env.SINGPASS_PAR_ENDPOINT!
  const authorizeEndpoint = process.env.SINGPASS_AUTHORIZE_ENDPOINT!
  const clientId = process.env.SINGPASS_CLIENT_ID!
  const redirectUri = process.env.SINGPASS_REDIRECT_URI!

  // Generate client assertion JWT (required for FAPI 2.0 private_key_jwt auth)
  const privateKey = await getPrivateKey()
  const { kid } = await getOrCreateKeyPair()

  const clientAssertion = await new SignJWT({})
    .setProtectedHeader({ alg: 'RS256', kid })
    .setIssuer(clientId)
    .setSubject(clientId)
    .setAudience(parEndpoint)
    .setExpirationTime('2m')
    .setJti(randomUUID())
    .setIssuedAt()
    .sign(privateKey)

  // Step 1: Push Authorization Request (PAR)
  const parParams = new URLSearchParams({
    response_type: 'code',
    client_id: clientId,
    scope: 'openid name dob residentialaddress nationality passtype workpassstatus',
    redirect_uri: redirectUri,
    code_challenge: codeChallenge,
    code_challenge_method: 'S256',
    state,
    nonce,
    client_assertion_type: 'urn:ietf:params:oauth:client-assertion-type:jwt-bearer',
    client_assertion: clientAssertion,
  })

  const parResponse = await fetch(parEndpoint, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: parParams,
  })

  if (!parResponse.ok) {
    throw new Error(`PAR request failed: ${parResponse.status} ${await parResponse.text()}`)
  }

  const { request_uri } = await parResponse.json() as { request_uri: string }

  // Store OAuth session for callback validation
  oauthSessions.set(sessionId, {
    codeVerifier,
    state,
    nonce,
    dpopPrivateKey,
    dpopPublicJWK,
    createdAt: Date.now(),
  })

  // Step 2: Return redirect URL — caller redirects user to SingPass
  return `${authorizeEndpoint}?client_id=${clientId}&request_uri=${encodeURIComponent(request_uri)}`
}

export function getOAuthSession(sessionId: string): OAuthSession | undefined {
  const session = oauthSessions.get(sessionId)
  // Clean up sessions older than 10 minutes
  if (session && Date.now() - session.createdAt > 10 * 60 * 1000) {
    oauthSessions.delete(sessionId)
    return undefined
  }
  return session
}

export function clearOAuthSession(sessionId: string): void {
  oauthSessions.delete(sessionId)
}
```
