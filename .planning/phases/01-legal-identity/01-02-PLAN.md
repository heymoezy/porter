```typescript
import { pgTable, uuid, text, boolean, timestamp, pgEnum } from 'drizzle-orm/pg-core'

export const onboardingStepEnum = pgEnum('onboarding_step', [
  'singpass_pending',
  'singpass_complete',
  'profile_complete',
  'industry_selected',
  'cert_pending',
  'cert_verified',
  'permit_verified',
  'contract_accepted',
  'complete',
])

export const industryEnum = pgEnum('worker_industry', ['fnb', 'retail', 'both'])

export const workerProfiles = pgTable('worker_profiles', {
  id: uuid('id').defaultRandom().primaryKey(),
  supabaseUserId: uuid('supabase_user_id').notNull().unique(),
  preferredName: text('preferred_name'),
  photoUrl: text('photo_url'),
  phoneNumber: text('phone_number'),
  industry: industryEnum('industry'),
  onboardingStep: onboardingStepEnum('onboarding_step').default('singpass_pending').notNull(),
  canBrowseShifts: boolean('can_browse_shifts').default(false).notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
})

export type WorkerProfile = typeof workerProfiles.$inferSelect
export type NewWorkerProfile = typeof workerProfiles.$inferInsert
```

Create `src/db/schema/verifications.ts`:
```typescript
import { pgTable, uuid, text, boolean, timestamp, pgEnum } from 'drizzle-orm/pg-core'
import { workerProfiles } from './workers'

export const verificationStatusEnum = pgEnum('verification_status', [
  'pending',
  'verified',
  'failed',
  'expired',
])

export const workerIdentityVerifications = pgTable('worker_identity_verifications', {
  id: uuid('id').defaultRandom().primaryKey(),
  workerId: uuid('worker_id').notNull().references(() => workerProfiles.id, { onDelete: 'cascade' }),
  // NEVER store raw NRIC — PDPA compliance. SHA-256(NRIC + salt) for dedup only.
  nricHash: text('nric_hash').notNull(),
  singpassVerifiedAt: timestamp('singpass_verified_at'),
  nationality: text('nationality'),   // 'SINGAPOREAN' | 'PR' | 'FOREIGNER'
  passType: text('pass_type'),        // 'S_PASS' | 'EP' | 'WP' | null
  nameVerified: boolean('name_verified').default(false).notNull(),
  dobVerified: boolean('dob_verified').default(false).notNull(),
  addressVerified: boolean('address_verified').default(false).notNull(),
  status: verificationStatusEnum('status').default('pending').notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

export const foodHygieneCerts = pgTable('food_hygiene_certs', {
  id: uuid('id').defaultRandom().primaryKey(),
  workerId: uuid('worker_id').notNull().references(() => workerProfiles.id, { onDelete: 'cascade' }),
  certImagePath: text('cert_image_path').notNull(), // Supabase Storage path
  expiryDate: timestamp('expiry_date').notNull(),
  verificationStatus: verificationStatusEnum('verification_status').default('pending').notNull(),
  reviewedAt: timestamp('reviewed_at'),
  reviewedBy: uuid('reviewed_by'), // admin user id
  autoBlockedAt: timestamp('auto_blocked_at'),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

// Work permit verification records for foreign workers
export const workPermitVerifications = pgTable('work_permit_verifications', {
  id: uuid('id').defaultRandom().primaryKey(),
  workerId: uuid('worker_id').notNull().references(() => workerProfiles.id, { onDelete: 'cascade' }),
  finNumber: text('fin_number').notNull(), // Foreign Identification Number
  permitType: text('permit_type'),         // 'S_PASS' | 'EP' | 'WP' | 'LTVP'
  permitStatus: text('permit_status'),     // 'Live' | 'Expired' | 'Cancelled'
  eligibleShiftCategories: text('eligible_shift_categories').array(), // ['fnb', 'retail']
  verifiedAt: timestamp('verified_at'),
  status: verificationStatusEnum('status').default('pending').notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

export type WorkerIdentityVerification = typeof workerIdentityVerifications.$inferSelect
export type FoodHygieneCert = typeof foodHygieneCerts.$inferSelect
export type WorkPermitVerification = typeof workPermitVerifications.$inferSelect
```

```typescript
import { pgTable, uuid, text, timestamp, integer } from 'drizzle-orm/pg-core'
import { workerProfiles } from './workers'
import { businessProfiles } from './businesses'

// Immutable contract versions — never updated, only new versions added
export const contractVersions = pgTable('contract_versions', {
  id: uuid('id').defaultRandom().primaryKey(),
  version: text('version').notNull().unique(), // e.g., '1.0', '1.1'
  contractText: text('contract_text').notNull(),
  effectiveDate: timestamp('effective_date').notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

// Worker accepts IC agreement
export const contractAcceptances = pgTable('contract_acceptances', {
  id: uuid('id').defaultRandom().primaryKey(),
  workerId: uuid('worker_id').notNull().references(() => workerProfiles.id, { onDelete: 'cascade' }),
  contractVersion: text('contract_version').notNull(),
  acceptedAt: timestamp('accepted_at').defaultNow().notNull(),
  ipAddress: text('ip_address'),
  userAgent: text('user_agent'),
})

// Business poaching clause versions
export const poachingClauseVersions = pgTable('poaching_clause_versions', {
  id: uuid('id').defaultRandom().primaryKey(),
  version: text('version').notNull().unique(),
  clauseText: text('clause_text').notNull(),
  conversionFeeFormula: text('conversion_fee_formula').notNull(), // '1 month salary = agreed hourly rate × 160 hours'
  effectiveDate: timestamp('effective_date').notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

// Business explicitly accepts poaching/conversion fee clause
export const businessPoachingAcceptances = pgTable('business_poaching_acceptances', {
  id: uuid('id').defaultRandom().primaryKey(),
  businessId: uuid('business_id').notNull().references(() => businessProfiles.id, { onDelete: 'cascade' }),
  clauseVersion: text('clause_version').notNull(),
  acceptedAt: timestamp('accepted_at').defaultNow().notNull(),
  ipAddress: text('ip_address'),
  userAgent: text('user_agent'),
})

export type ContractAcceptance = typeof contractAcceptances.$inferSelect
export type BusinessPoachingAcceptance = typeof businessPoachingAcceptances.$inferSelect
```
