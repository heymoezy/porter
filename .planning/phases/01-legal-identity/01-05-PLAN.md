export interface UENVerificationResult {
  valid: boolean
  entityName?: string
  registrationDate?: string
  businessActivity?: string
  reason?: 'uen_not_found' | 'entity_not_active' | 'api_unavailable'
}

export async function verifyUEN(uen: string): Promise<UENVerificationResult> {
  // Validate UEN format: 9-10 alphanumeric characters
  if (!/^[0-9A-Z]{9,10}$/.test(uen.toUpperCase())) {
    return { valid: false, reason: 'uen_not_found' }
  }

  // Option A: ACRA BizFile Business Profile API (preferred — launched Nov 2025)
  // Register at bizfile.gov.sg/apimarketplace (requires CorpPass)
  // See RESEARCH.md AUTH-05 notes
  const acraApiBase = process.env.ACRA_API_BASE
  const acraApiToken = process.env.ACRA_API_TOKEN

  if (acraApiBase && acraApiToken) {
    try {
      const response = await fetch(`${acraApiBase}/business-profile/${uen.toUpperCase()}`, {
        headers: { Authorization: `Bearer ${acraApiToken}` },
        signal: AbortSignal.timeout(5000), // 5s timeout
      })

      if (response.status === 404) {
        return { valid: false, reason: 'uen_not_found' }
      }

      if (!response.ok) {
        // API error — fall through to local dataset
        return await verifyUENFromLocalDataset(uen)
      }

      const profile = await response.json()
      const isActive = profile.status === 'Active' || profile.status === 'Live'

      if (!isActive) {
        return { valid: false, reason: 'entity_not_active' }
      }

      return {
        valid: true,
        entityName: profile.entity_name,
        registrationDate: profile.reg_date,
        businessActivity: profile.primary_ssic_description,
      }
    } catch (error) {
      // Network/timeout error — fall through to local dataset
    }
  }

  // Option B: Local dataset fallback (data.gov.sg ACRA monthly export)
  return verifyUENFromLocalDataset(uen)
}

// Fallback: local ACRA dataset (monthly CSV from data.gov.sg)
// Not real-time but sufficient for MVP — see RESEARCH.md AUTH-05
async function verifyUENFromLocalDataset(uen: string): Promise<UENVerificationResult> {
  // TODO: Load from local JSON file pre-populated from data.gov.sg ACRA export
  // Until the dataset file is loaded, return api_unavailable
  const datasetPath = process.env.ACRA_LOCAL_DATASET_PATH
  if (!datasetPath) {
    return { valid: false, reason: 'api_unavailable' }
  }

  try {
    const { readFileSync } = await import('fs')
    const dataset = JSON.parse(readFileSync(datasetPath, 'utf-8'))
    const entry = dataset[uen.toUpperCase()]
    if (!entry) return { valid: false, reason: 'uen_not_found' }
    if (entry.status !== 'Active' && entry.status !== 'Live') {
      return { valid: false, reason: 'entity_not_active' }
    }
    return { valid: true, entityName: entry.entity_name, businessActivity: entry.ssic }
  } catch {
    return { valid: false, reason: 'api_unavailable' }
  }
}

import { businessWaitlist } from '@/db/schema/businesses'

interface ClusterDefinition {
  name: string
  postalPrefixes: string[]
}

// Launch clusters — initial deployment in Tanjong Pagar / CBD district
// CONTEXT.md: "researcher should identify densest F&B + retail cluster"
// Current: using Tanjong Pagar / CBD (postal prefixes 01-08) as starting point
// Validate against SFA licensed food establishment data at data.gov.sg before launch
export const LAUNCH_CLUSTERS: ClusterDefinition[] = [
  {
    name: 'Tanjong Pagar / CBD',
    postalPrefixes: ['01', '02', '03', '04', '05', '06', '07', '08'],
  },
]

export function getLaunchClusters(): ClusterDefinition[] {
  return LAUNCH_CLUSTERS
}

export function isWithinLaunchCluster(postalCode: string): boolean {
  const prefix = postalCode.substring(0, 2)
  return LAUNCH_CLUSTERS.some(cluster => cluster.postalPrefixes.includes(prefix))
}

// Record a business outside the cluster in the waitlist for future notification
export async function addToBusinessWaitlist(params: {
  uen?: string
  entityName?: string
  postalCode: string
  contactEmail?: string
}): Promise<void> {
  await db.insert(businessWaitlist).values({
    uen: params.uen,
    entityName: params.entityName,
    postalCode: params.postalCode,
    contactEmail: params.contactEmail,
  })
}

import { db } from '@/db'
import { contractAcceptances, businessPoachingAcceptances } from '@/db/schema/contracts'
import { eq, and } from 'drizzle-orm'

// Record worker acceptance of the independent contractor agreement
// Called at onboarding Step 6 — after user explicitly checks the "I have read and agree" checkbox
export async function recordWorkerContractAcceptance(
  workerId: string,
  contractVersion: string,
  ipAddress?: string,
  userAgent?: string,
): Promise<void> {
  await db.insert(contractAcceptances).values({
    workerId,
    contractVersion,
    acceptedAt: new Date(),
    ipAddress,
    userAgent,
  })
}

// Check if worker has accepted the current contract version
// If version has changed, worker must re-accept before their next shift bid
export async function hasAcceptedCurrentContract(
  workerId: string,
  currentVersion: string,
): Promise<boolean> {
  const acceptance = await db.query.contractAcceptances.findFirst({
    where: and(
      eq(contractAcceptances.workerId, workerId),
      eq(contractAcceptances.contractVersion, currentVersion),
    ),
  })
  return acceptance !== undefined
}

// Record business acceptance of the poaching/conversion fee clause
// CONTEXT.md locked: must be a SEPARATE screen with its own checkbox
// Pitfall 4: buried-in-ToS poaching clauses are not enforceable
export async function recordBusinessPoachingAcceptance(
  businessId: string,
  clauseVersion: string,
  ipAddress?: string,
  userAgent?: string,
): Promise<void> {
  await db.insert(businessPoachingAcceptances).values({
    businessId,
    clauseVersion,
    acceptedAt: new Date(),
    ipAddress,
    userAgent,
  })
}

// Check if business has accepted the current poaching clause version
// Business cannot post shifts without this acceptance
export async function hasAcceptedCurrentPoachingClause(
  businessId: string,
  currentClauseVersion: string,
): Promise<boolean> {
  const acceptance = await db.query.businessPoachingAcceptances.findFirst({
    where: and(
      eq(businessPoachingAcceptances.businessId, businessId),
      eq(businessPoachingAcceptances.clauseVersion, currentClauseVersion),
    ),
  })
  return acceptance !== undefined
}

// Get the most recent worker contract acceptance (any version)
export async function getLatestWorkerContractAcceptance(workerId: string) {
  return db.query.contractAcceptances.findFirst({
    where: eq(contractAcceptances.workerId, workerId),
    orderBy: (acc, { desc }) => [desc(acc.acceptedAt)],
  })
}

// Insurance activation — per-shift WICA-equivalent coverage
// RESEARCH.md Pitfall 6: insurance must be actually activated per shift, not just "baked into markup"
// Approach: aggregate policy covers all confirmed shifts. insurance_covered=true is the activation record.
import { db } from '@/db'
import { shiftInsuranceCoverage, insurancePolicies } from '@/db/schema/insurance'
import { eq, and } from 'drizzle-orm'

// Called at shift confirmation — sets insurance_covered=true
// The shift record becomes the evidence of coverage for WICA claims
export async function activateShiftInsurance(
  shiftId: string,
  workerId: string,
  insurancePolicyId?: string,
): Promise<void> {
  await db.insert(shiftInsuranceCoverage).values({
    shiftId,
    workerId,
    insurancePolicyId,
    insuranceCovered: true,
    activatedAt: new Date(),
  })
}

// Called at shift completion or confirmed cancellation before start
export async function deactivateShiftInsurance(shiftId: string): Promise<void> {
  await db
    .update(shiftInsuranceCoverage)
    .set({ deactivatedAt: new Date() })
    .where(eq(shiftInsuranceCoverage.shiftId, shiftId))
}

// Get the active insurance policy (most recent active policy)
export async function getActiveInsurancePolicy() {
  return db.query.insurancePolicies.findFirst({
    where: eq(insurancePolicies.isActive, true),
    orderBy: (policies, { desc }) => [desc(policies.coverageStartDate)],
  })
}

// Check if a shift has insurance coverage
export async function isShiftInsured(shiftId: string): Promise<boolean> {
  const coverage = await db.query.shiftInsuranceCoverage.findFirst({
    where: and(
      eq(shiftInsuranceCoverage.shiftId, shiftId),
      eq(shiftInsuranceCoverage.insuranceCovered, true),
    ),
  })
  return coverage !== undefined
}
