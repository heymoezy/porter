```typescript
// POST /api/auth/business-onboarding
// Handles all 5 steps of business onboarding — validates and gates each step
import { NextRequest, NextResponse } from 'next/server'
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import { db } from '@/db'
import { businessProfiles } from '@/db/schema/businesses'
import { eq } from 'drizzle-orm'
import { verifyUEN } from '@/services/identity/uen-verifier'
import { isWithinLaunchCluster, addToBusinessWaitlist } from '@/services/identity/cluster-gate'
import { recordBusinessPoachingAcceptance } from '@/services/identity/contract'

const CURRENT_POACHING_CLAUSE_VERSION = process.env.CURRENT_POACHING_CLAUSE_VERSION ?? '1.0'

export async function POST(request: NextRequest) {
  const cookieStore = await cookies()
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { cookies: { getAll: () => cookieStore.getAll() } }
  )

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const body = await request.json() as Record<string, unknown>
  const { step } = body

  switch (step) {
    case 'uen': {
      const { uen } = body as { uen: string }
      if (!uen || typeof uen !== 'string') {
        return NextResponse.json({ error: 'UEN is required' }, { status: 400 })
      }

      const result = await verifyUEN(uen.trim().toUpperCase())
      if (!result.valid) {
        return NextResponse.json(
          { error: result.reason ?? 'uen_not_found', valid: false },
          { status: 400 }
        )
      }

      // Store verified UEN details in session or temp state
      // Create initial business profile record if not exists
      const existing = await db.query.businessProfiles.findFirst({
        where: eq(businessProfiles.supabaseUserId, user.id),
      })

      if (!existing) {
        await db.insert(businessProfiles).values({
          supabaseUserId: user.id,
          uen: uen.trim().toUpperCase(),
          entityName: result.entityName ?? uen,
          industry: 'fnb', // default — updated at profile step
          postalCode: '',   // filled at profile step
          uenVerifiedAt: new Date(),
          canPostShifts: false,
          status: 'pending',
        })
      } else {
        await db.update(businessProfiles).set({
          uen: uen.trim().toUpperCase(),
          entityName: result.entityName ?? existing.entityName,
          uenVerifiedAt: new Date(),
        }).where(eq(businessProfiles.supabaseUserId, user.id))
      }

      return NextResponse.json({
        ok: true,
        entityName: result.entityName,
        businessActivity: result.businessActivity,
        nextStep: 'profile',
      })
    }

    case 'profile': {
      const { entityName, industry, postalCode, address, contactEmail, contactPhone } = body as {
        entityName: string
        industry: 'fnb' | 'retail' | 'both'
        postalCode: string
        address?: string
        contactEmail?: string
        contactPhone?: string
      }

      if (!entityName || !industry || !postalCode) {
        return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
      }

      await db.update(businessProfiles).set({
        entityName,
        industry,
        postalCode,
        address,
        contactEmail,
        contactPhone,
      }).where(eq(businessProfiles.supabaseUserId, user.id))

      return NextResponse.json({ ok: true, nextStep: 'cluster' })
    }

    case 'cluster': {
      const business = await db.query.businessProfiles.findFirst({
        where: eq(businessProfiles.supabaseUserId, user.id),
      })

      if (!business) {
        return NextResponse.json({ error: 'Business profile not found' }, { status: 404 })
      }

      const inCluster = isWithinLaunchCluster(business.postalCode)

      if (!inCluster) {
        // Add to waitlist and block onboarding
        await addToBusinessWaitlist({
          uen: business.uen,
          entityName: business.entityName,
          postalCode: business.postalCode,
          contactEmail: business.contactEmail ?? undefined,
        })
        return NextResponse.json({
          ok: false,
          blocked: true,
          reason: 'outside_launch_cluster',
          message: 'ShiftSG is currently only available in the Tanjong Pagar / CBD area. We\'ve added you to the waitlist.',
        })
      }

      await db.update(businessProfiles).set({
        clusterVerifiedAt: new Date(),
      }).where(eq(businessProfiles.supabaseUserId, user.id))

      return NextResponse.json({ ok: true, nextStep: 'poaching' })
    }

    case 'poaching': {
      const { accepted } = body as { accepted: boolean }
      if (!accepted) {
        return NextResponse.json({ error: 'Poaching clause not accepted' }, { status: 400 })
      }

      const business = await db.query.businessProfiles.findFirst({
        where: eq(businessProfiles.supabaseUserId, user.id),
      })

      if (!business) {
        return NextResponse.json({ error: 'Business profile not found' }, { status: 404 })
      }

      const ipAddress = request.headers.get('x-forwarded-for')?.split(',')[0] ?? undefined
      const userAgent = request.headers.get('user-agent') ?? undefined

      await recordBusinessPoachingAcceptance(
        business.id,
        CURRENT_POACHING_CLAUSE_VERSION,
        ipAddress,
        userAgent,
      )

      // CONTEXT.md Step 5: Payment setup is the final gate before can_post_shifts=true
      // Payment setup is NOT implemented in Phase 1 (deferred — needs Stripe/payment partner research)
      // Phase 1 sets canPostShifts=false with status=pending_payment so the business can proceed
      // to the complete screen but cannot post shifts until Phase 2 payment setup is done.
      // This is intentional: CONTEXT.md says "Step 5: Payment setup" before "Step 6: Onboarding complete"
      await db.update(businessProfiles).set({
        canPostShifts: false,    // NOT true — payment setup required first (Phase 2)
        status: 'pending_payment', // Signal to Phase 2 that this business needs payment setup
      }).where(eq(businessProfiles.id, business.id))

      return NextResponse.json({
        ok: true,
        nextStep: 'payment_setup', // Phase 2 completes this step
        message: 'Poaching clause accepted. Complete payment setup to start posting shifts.',
      })
    }

    default:
      return NextResponse.json({ error: 'Unknown step' }, { status: 400 })
  }
}
```

```tsx
// Step 1: UEN verification
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'

export default function BusinessUenPage() {
  const router = useRouter()
  const [uen, setUen] = useState('')
  const [isVerifying, setIsVerifying] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [result, setResult] = useState<{ entityName?: string; businessActivity?: string } | null>(null)

  const handleVerify = async (e: React.FormEvent) => {
    e.preventDefault()
    setIsVerifying(true)
    setError(null)

    const res = await fetch('/api/auth/business-onboarding', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ step: 'uen', uen }),
    })

    const data = await res.json()

    if (!res.ok) {
      const errorMessages: Record<string, string> = {
        uen_not_found: 'UEN not found. Please check the number and try again.',
        entity_not_active: 'This business is no longer active with ACRA.',
        api_unavailable: 'UEN verification is temporarily unavailable. Please try again.',
      }
      setError(errorMessages[data.error] ?? 'Verification failed. Please try again.')
      setIsVerifying(false)
      return
    }

    setResult({ entityName: data.entityName, businessActivity: data.businessActivity })
    // Auto-proceed to profile step after brief confirmation
    setTimeout(() => router.push('/business/profile'), 1500)
  }

  return (
    <div className="flex min-h-screen flex-col items-center justify-center px-4">
      <div className="w-full max-w-md space-y-6">
        <div>
          <h1 className="text-2xl font-bold text-gray-900">Verify your business</h1>
          <p className="mt-2 text-gray-600">
            Enter your UEN (Unique Entity Number) to verify your business with ACRA.
          </p>
        </div>

        {!result ? (
          <form onSubmit={handleVerify} className="space-y-4">
            <div>
              <label className="block text-sm font-medium text-gray-700">
                UEN (Unique Entity Number)
              </label>
              <input
                type="text"
                value={uen}
                onChange={e => setUen(e.target.value.toUpperCase())}
                required
                pattern="[0-9A-Z]{9,10}"
                placeholder="e.g. 202012345A"
                className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 font-mono uppercase"
              />
              <p className="mt-1 text-xs text-gray-500">
                Find your UEN on ACRA BizFile or your incorporation documents.
              </p>
            </div>

            {error && (
              <div className="rounded-md bg-red-50 p-4 text-sm text-red-700">{error}</div>
            )}

            <button
              type="submit"
              disabled={isVerifying || uen.length < 9}
              className="w-full rounded-lg bg-blue-600 px-6 py-3 text-white font-medium
                         hover:bg-blue-700 disabled:opacity-50"
            >
              {isVerifying ? 'Verifying…' : 'Verify UEN'}
            </button>
          </form>
        ) : (
          <div className="rounded-lg bg-green-50 border border-green-200 p-4">
            <p className="font-medium text-green-800">Verified</p>
            <p className="text-green-700">{result.entityName}</p>
            {result.businessActivity && (
              <p className="text-sm text-green-600">{result.businessActivity}</p>
            )}
            <p className="mt-2 text-sm text-green-600">Proceeding to profile setup…</p>
          </div>
        )}
      </div>
    </div>
  )
}
```

```tsx
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'

export default function BusinessProfilePage() {
  const router = useRouter()
  const [industry, setIndustry] = useState<'fnb' | 'retail' | 'both'>('fnb')
  const [postalCode, setPostalCode] = useState('')
  const [address, setAddress] = useState('')
  const [contactEmail, setContactEmail] = useState('')
  const [contactPhone, setContactPhone] = useState('')
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setIsSubmitting(true)
    setError(null)

    // First save profile
    const profileRes = await fetch('/api/auth/business-onboarding', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ step: 'profile', industry, postalCode, address, contactEmail, contactPhone }),
    })

    if (!profileRes.ok) {
      setError('Failed to save profile. Please try again.')
      setIsSubmitting(false)
      return
    }

    // Then run cluster check
    const clusterRes = await fetch('/api/auth/business-onboarding', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ step: 'cluster' }),
    })

    const clusterData = await clusterRes.json()

    if (clusterData.blocked) {
      router.push('/business/cluster-blocked')
      return
    }

    if (!clusterRes.ok) {
      setError('Something went wrong. Please try again.')
      setIsSubmitting(false)
      return
    }

    router.push('/business/poaching-clause')
  }

  return (
    <div className="flex min-h-screen flex-col items-center justify-center px-4">
      <div className="w-full max-w-md space-y-6">
        <h1 className="text-2xl font-bold text-gray-900">Business details</h1>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-gray-700">Industry</label>
            <select
              value={industry}
              onChange={e => setIndustry(e.target.value as 'fnb' | 'retail' | 'both')}
              className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2"
            >
              <option value="fnb">Food & Beverage</option>
              <option value="retail">Retail</option>
              <option value="both">Both</option>
            </select>
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700">Postal code</label>
            <input
              type="text"
              value={postalCode}
              onChange={e => setPostalCode(e.target.value)}
              required
              pattern="\d{6}"
              maxLength={6}
              placeholder="018989"
              className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 font-mono"
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700">Address</label>
            <input
              type="text"
              value={address}
              onChange={e => setAddress(e.target.value)}
              placeholder="1 Tanjong Pagar Plaza, #01-01"
              className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2"
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700">Contact email</label>
            <input
              type="email"
              value={contactEmail}
              onChange={e => setContactEmail(e.target.value)}
              required
              className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2"

          </div>

            type="submit"

  )
}
```
