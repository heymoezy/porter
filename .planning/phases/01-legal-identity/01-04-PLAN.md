import { db } from '@/db'
import { foodHygieneCerts } from '@/db/schema/verifications'
import { workerProfiles } from '@/db/schema/workers'
import { eq, and, gt, lte, gte } from 'drizzle-orm'
import { createClient } from '@supabase/supabase-js'

// Upload a cert image to Supabase Storage and create a foodHygieneCerts record
// Returns the storage path (used by admin cert review queue to display the image)
export async function uploadCert(params: {
  workerId: string
  file: File | Blob
  fileName: string
  expiryDate: Date
}): Promise<string> {
  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!, // service role bypasses RLS for server-side upload
  )

  // Upload to Supabase Storage: food-hygiene-certs/{workerId}/{timestamp}-{fileName}
  const storagePath = `food-hygiene-certs/${params.workerId}/${Date.now()}-${params.fileName}`

  const { error: uploadError } = await supabase.storage
    .from('worker-certs')
    .upload(storagePath, params.file, {
      contentType: params.file.type ?? 'image/jpeg',
      upsert: false,
    })

  if (uploadError) {
    throw new Error(`Cert upload to Supabase Storage failed: ${uploadError.message}`)
  }

  // Insert cert record with status=pending (awaiting admin review)
  await db.insert(foodHygieneCerts).values({
    workerId: params.workerId,
    certImagePath: storagePath,
    expiryDate: params.expiryDate,
    verificationStatus: 'pending',
  })

  return storagePath
}

// Check if a worker is eligible for F&B shifts
// Two-layer defense (RESEARCH.md Pitfall 5):
// Layer 1: This function checks real-time cert status
// Layer 2: Shift matching query ALSO filters expiry_date > NOW() at query time
export async function isEligibleForFnbShifts(workerId: string): Promise<boolean> {
  const cert = await db.query.foodHygieneCerts.findFirst({
    where: and(
      eq(foodHygieneCerts.workerId, workerId),
      eq(foodHygieneCerts.verificationStatus, 'verified'),
      gt(foodHygieneCerts.expiryDate, new Date()),
    ),
  })
  return cert !== undefined
}

// Get the most recent cert record for a worker (any status)
export async function getCertStatus(workerId: string) {
  const cert = await db.query.foodHygieneCerts.findFirst({
    where: eq(foodHygieneCerts.workerId, workerId),
    orderBy: (certs, { desc }) => [desc(certs.createdAt)],
  })
  return cert ?? null
}

// Called when admin manually approves a cert in the review queue
export async function approveCert(certId: string, reviewedByAdminId: string): Promise<void> {
  await db
    .update(foodHygieneCerts)
    .set({
      verificationStatus: 'verified',
      reviewedAt: new Date(),
      reviewedBy: reviewedByAdminId,
    })
    .where(eq(foodHygieneCerts.id, certId))
}

// Called when admin rejects a cert
export async function rejectCert(certId: string, reviewedByAdminId: string): Promise<void> {
  await db
    .update(foodHygieneCerts)
    .set({
      verificationStatus: 'failed',
      reviewedAt: new Date(),
      reviewedBy: reviewedByAdminId,
    })
    .where(eq(foodHygieneCerts.id, certId))
}

// Mark cert as expired — called by Inngest job on expiry date
export async function expireCert(certId: string): Promise<void> {
  await db
    .update(foodHygieneCerts)
    .set({
      verificationStatus: 'expired',
      autoBlockedAt: new Date(),
    })
    .where(eq(foodHygieneCerts.id, certId))
}

// Find certs expiring within N days (verified only — pending/failed/expired not relevant)
export async function findCertsExpiringSoon(withinDays: number) {
  const cutoff = new Date()
  cutoff.setDate(cutoff.getDate() + withinDays)
  const today = new Date()

  return db.query.foodHygieneCerts.findMany({
    where: and(
      eq(foodHygieneCerts.verificationStatus, 'verified'),
      lte(foodHygieneCerts.expiryDate, cutoff),
      gte(foodHygieneCerts.expiryDate, today),
    ),
  })
}

// Find certs that have passed their expiry date but not yet marked expired
export async function findExpiredCerts() {
  return db.query.foodHygieneCerts.findMany({
    where: and(
      eq(foodHygieneCerts.verificationStatus, 'verified'),
      lte(foodHygieneCerts.expiryDate, new Date()),
    ),
  })
}

import { db } from '@/db'
import { workPermitVerifications } from '@/db/schema/verifications'
import { eq } from 'drizzle-orm'

// FIN format: 1 letter + 7 digits + 1 letter (e.g., G1234567K)
// NRIC format (citizens/PR): S or T + 7 digits + 1 letter (e.g., S1234567D)
const FIN_REGEX = /^[A-Z]\d{7}[A-Z]$/

export function validateFin(fin: string): { valid: boolean; reason?: string } {
  if (!FIN_REGEX.test(fin.toUpperCase())) {
    return { valid: false, reason: 'invalid_format' }
  }
  return { valid: true }
}

// Determine permitted shift categories based on nationality and pass type
// Sources: MOM work pass framework + CONTEXT.md locked decisions
export function getPermittedShiftCategories(
  nationality: string,
  passType: string | null,
): string[] {
  // Singapore citizens and PRs have no work restrictions
  if (nationality === 'SINGAPOREAN' || nationality === 'PR') {
    return ['fnb', 'retail']
  }

  // Foreign workers — categorize by pass type
  // S Pass and Employment Pass: no sector restrictions for eligible jobs
  if (passType === 'S_PASS' || passType === 'EP') {
    return ['fnb', 'retail']
  }

  // Work Permit: allowed in F&B and retail (common WP categories in Singapore)
  // Note: specific sector restrictions depend on MOM approval for individual worker
  // At MVP, we store the permit type and flag for manual review if needed
  if (passType === 'WP') {
    return ['fnb', 'retail']
  }

  // Long-Term Visit Pass: not eligible for paid work without approval
  // Return empty array — no shift categories permitted until verified
  if (passType === 'LTVP') {
    return []
  }

  // Unknown pass type — restrict until manually verified
  return []
}

// Check if a worker's permit is currently valid
export async function checkWorkPermitEligibility(workerId: string): Promise<{
  requiresPermitCheck: boolean
  isVerified: boolean
  permittedCategories: string[]
}> {
  const permit = await db.query.workPermitVerifications.findFirst({
    where: eq(workPermitVerifications.workerId, workerId),
    orderBy: (permits, { desc }) => [desc(permits.createdAt)],
  })

  if (!permit) {
    return { requiresPermitCheck: true, isVerified: false, permittedCategories: [] }
  }

  if (permit.status === 'verified' && permit.eligibleShiftCategories) {
    return {
      requiresPermitCheck: false,
      isVerified: true,
      permittedCategories: permit.eligibleShiftCategories,
    }
  }

  return { requiresPermitCheck: true, isVerified: false, permittedCategories: [] }
}

// Store work permit verification result from MOM check
// At MVP: manual check — admin confirms FIN + permit number match MOM records
// Future: MOM API integration if available
export async function recordWorkPermitVerification(
  workerId: string,
  finNumber: string,
  permitType: string,
  nationality: string,
): Promise<void> {
  const permittedCategories = getPermittedShiftCategories(nationality, permitType)

  await db.insert(workPermitVerifications).values({
    workerId,
    finNumber,
    permitType,
    permitStatus: 'Live',
    eligibleShiftCategories: permittedCategories,
    verifiedAt: new Date(),
    status: 'verified',
  })
}

// Inngest scheduled function — cert expiry warning and auto-block
// Source: RESEARCH.md Inngest example (Pattern: Don't Hand-Roll)
// Runs daily at 9am SGT (UTC+8 = 01:00 UTC)

import { inngest } from '@/inngest/client'
import { findCertsExpiringSoon, findExpiredCerts, expireCert } from '@/services/identity/cert-tracker'

// Placeholder for push notification service (Phase 2 wires the actual notification channel)
async function sendCertExpiryWarning(workerId: string, daysLeft: number): Promise<void> {
  // TODO Phase 2: integrate with push notification service
  // For now, log to DB or no-op — notifications handled via email at MVP
  console.log(`[cert-expiry-check] Worker ${workerId} cert expires in ${daysLeft} days`)
}

export const certExpiryCheck = inngest.createFunction(
  { id: 'cert-expiry-check', name: 'Daily FHD2H Cert Expiry Check' },
  { cron: '0 1 * * *' }, // 01:00 UTC = 09:00 SGT
  async ({ step }) => {
    // Step 1: Find certs expiring in 30 days — send warning
    const expiring30 = await step.run('find-certs-expiring-30d', async () => {
      return findCertsExpiringSoon(30)
    })

    await step.run('notify-30d-warnings', async () => {
      for (const cert of expiring30) {
        const daysLeft = Math.ceil(
          (cert.expiryDate.getTime() - Date.now()) / (1000 * 60 * 60 * 24)
        )
        await sendCertExpiryWarning(cert.workerId, daysLeft)
      }
    })

    // Step 2: Find certs expiring in 7 days — send urgent warning
    const expiring7 = await step.run('find-certs-expiring-7d', async () => {
      return findCertsExpiringSoon(7)
    })

    await step.run('notify-7d-warnings', async () => {
      for (const cert of expiring7) {
        const daysLeft = Math.ceil(
          (cert.expiryDate.getTime() - Date.now()) / (1000 * 60 * 60 * 24)
        )
        await sendCertExpiryWarning(cert.workerId, daysLeft)
      }
    })

    // Step 3: Find certs that have expired — mark them expired (blocks F&B shift access)
    const expired = await step.run('find-expired-certs', async () => {
      return findExpiredCerts()
    })

    await step.run('expire-certs', async () => {
      for (const cert of expired) {
        await expireCert(cert.id)
      }
    })

    return {
      notified30d: expiring30.length,
      notified7d: expiring7.length,
      expired: expired.length,
    }
  },
)
