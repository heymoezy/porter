```typescript
// GET /api/admin/certs
// Returns all pending cert submissions for admin review
// Admin-only endpoint — requires elevated session in production
import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/db'
import { foodHygieneCerts } from '@/db/schema/verifications'
import { workerProfiles } from '@/db/schema/workers'
import { eq } from 'drizzle-orm'

export async function GET(request: NextRequest) {
  // TODO: Add admin auth check before going to production
  // For MVP: simple admin token from env var
  const adminToken = request.headers.get('x-admin-token')
  if (adminToken !== process.env.ADMIN_API_TOKEN) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const pendingCerts = await db.query.foodHygieneCerts.findMany({
    where: eq(foodHygieneCerts.verificationStatus, 'pending'),
    orderBy: (certs, { asc }) => [asc(certs.createdAt)],
  })

  return NextResponse.json({ certs: pendingCerts })
}
```

```typescript
// PATCH /api/admin/certs/:certId
// Approve or reject a cert submission
import { NextRequest, NextResponse } from 'next/server'
import { approveCert, rejectCert } from '@/services/identity/cert-tracker'

export async function PATCH(
  request: NextRequest,
  { params }: { params: { certId: string } }
) {
  const adminToken = request.headers.get('x-admin-token')
  if (adminToken !== process.env.ADMIN_API_TOKEN) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const { certId } = params
  const { action, reviewedBy } = await request.json() as {
    action: 'approve' | 'reject'
    reviewedBy: string // admin user ID
  }

  if (!certId || !action || !reviewedBy) {
    return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
  }

  if (action === 'approve') {
    await approveCert(certId, reviewedBy)
    return NextResponse.json({ ok: true, action: 'approved' })
  } else if (action === 'reject') {
    await rejectCert(certId, reviewedBy)
    return NextResponse.json({ ok: true, action: 'rejected' })
  } else {
    return NextResponse.json({ error: 'Invalid action' }, { status: 400 })
  }
}
```

```tsx
// Admin cert review queue — simple table of pending submissions
// Internal tool — not public-facing
'use client'

import { useEffect, useState } from 'react'

interface CertRecord {
  id: string
  workerId: string
  certImagePath: string
  expiryDate: string
  verificationStatus: string
  createdAt: string
}

export default function AdminCertQueuePage() {
  const [certs, setCerts] = useState<CertRecord[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [adminToken] = useState(process.env.NEXT_PUBLIC_ADMIN_TOKEN ?? '')

  useEffect(() => {
    fetchPendingCerts()
  }, [])

  const fetchPendingCerts = async () => {
    const res = await fetch('/api/admin/certs', {
      headers: { 'x-admin-token': adminToken },
    })
    if (res.ok) {
      const data = await res.json()
      setCerts(data.certs)
    }
    setIsLoading(false)
  }

  const handleAction = async (certId: string, action: 'approve' | 'reject') => {
    const res = await fetch(`/api/admin/certs/${certId}`, {
      method: 'PATCH',
      headers: {
        'Content-Type': 'application/json',
        'x-admin-token': adminToken,
      },
      body: JSON.stringify({ action, reviewedBy: 'admin' }),
    })
    if (res.ok) {
      setCerts(prev => prev.filter(c => c.id !== certId))
    }
  }

  if (isLoading) return <div className="p-8 text-gray-500">Loading pending certifications…</div>

  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold text-gray-900 mb-6">
        Food Hygiene Cert Review Queue
      </h1>

      {certs.length === 0 ? (
        <p className="text-gray-500">No pending certifications.</p>
      ) : (
        <div className="space-y-4">
          <p className="text-sm text-gray-600">{certs.length} pending review</p>
          {certs.map(cert => (
            <div
              key={cert.id}
              className="rounded-lg border border-gray-200 bg-white p-4 flex items-start justify-between"
            >
              <div className="space-y-1">
                <p className="text-sm font-medium text-gray-900">Worker: {cert.workerId}</p>
                <p className="text-sm text-gray-600">
                  Expires: {new Date(cert.expiryDate).toLocaleDateString('en-SG')}
                </p>
                <p className="text-sm text-gray-600">
                  Submitted: {new Date(cert.createdAt).toLocaleDateString('en-SG')}
                </p>
                <a
                  href={cert.certImagePath}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="text-sm text-blue-600 hover:underline"
                >
                  View certificate
                </a>
              </div>
              <div className="flex space-x-2">
                <button
                  onClick={() => handleAction(cert.id, 'approve')}
                  className="rounded-md bg-green-600 px-4 py-2 text-sm text-white hover:bg-green-700"
                >
                  Approve
                </button>
                <button
                  onClick={() => handleAction(cert.id, 'reject')}
                  className="rounded-md bg-red-600 px-4 py-2 text-sm text-white hover:bg-red-700"
                >
                  Reject
                </button>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
```

```tsx
'use client'

import { useEffect, useState } from 'react'
import { useRouter } from 'next/navigation'

interface WorkerStatus {
  onboardingStep: string
  canBrowseShifts: boolean
  industry: string | null
}

export default function WorkerCompletePage() {
  const router = useRouter()
  const [status, setStatus] = useState<WorkerStatus | null>(null)
  const [isLoading, setIsLoading] = useState(true)

  useEffect(() => {
    fetch('/api/auth/worker-onboarding/status')
      .then(r => r.json())
      .then(data => { setStatus(data); setIsLoading(false) })
      .catch(() => setIsLoading(false))
  }, [])

  if (isLoading) return <div className="flex min-h-screen items-center justify-center text-gray-500">Loading…</div>

  const hasPendingCert = status?.onboardingStep === 'cert_pending'
  const isFnbWorker = status?.industry === 'fnb' || status?.industry === 'both'

  return (
    <div className="flex min-h-screen flex-col items-center justify-center px-4">
      <div className="w-full max-w-md space-y-6 text-center">
        <div className="text-4xl">&#10003;</div>
```
