Create `src/lib/test-utils/mockpass-setup.ts`:
```typescript
import { spawn, ChildProcess } from 'child_process'

let mockpassProcess: ChildProcess | null = null

export async function startMockPass(port = 5156): Promise<void> {
  return new Promise((resolve, reject) => {
    mockpassProcess = spawn('npx', ['mockpass', '--port', String(port)], {
      env: { ...process.env, MOCKPASS_PORT: String(port) },
      stdio: ['ignore', 'pipe', 'pipe'],
    })
    mockpassProcess.stdout?.on('data', (data: Buffer) => {
      if (data.toString().includes('MockPass')) resolve()
    })
    mockpassProcess.stderr?.on('data', (data: Buffer) => {
      if (data.toString().includes('Error')) reject(new Error(data.toString()))
    })
    setTimeout(resolve, 3000) // fallback if no stdout signal
  })
}

export async function stopMockPass(): Promise<void> {
  mockpassProcess?.kill('SIGTERM')
  mockpassProcess = null
}
```
