Create `backend/scripts/migrate-skills-sot.ts` — a one-shot migration script run via `npx tsx backend/scripts/migrate-skills-sot.ts`.

The script must:

**Step 1: Apply DDL** (idempotent)
```sql
ALTER TABLE persona_skills ADD COLUMN IF NOT EXISTS skill_id TEXT;
```

**Step 2: Populate template_skills from agent_templates.skills JSONB**

For each template row:
1. Parse `skills` JSONB array (e.g., `["frontend", "react", "typescript"]`)
2. For each skill tag, attempt to match to a `skills.id`:
   - Exact match: `SELECT id FROM skills WHERE id = $tag`
   - Fuzzy match: `SELECT id FROM skills WHERE id LIKE '%' || $tag || '%' LIMIT 1`
   - If no match found: log warning and skip (do NOT create phantom skills rows)
3. INSERT matched skill IDs into template_skills with sort_order = array index
4. Use `ON CONFLICT DO NOTHING` on (template_id, skill_id) to be idempotent

**Step 3: Migrate persona_skills.skill_id for existing 17 porter-core rows**

For each persona_skills row where skill_id IS NULL:
1. Look up `skills.id` matching `skill_name` — try exact match first, then `skills.id LIKE '%' || skill_name || '%'`
2. If matched, `UPDATE persona_skills SET skill_id = $matched WHERE persona_id = $pid AND skill_name = $sname`
3. Log unmatched rows

**Step 4: Report**
Print summary:
- Templates processed: N
- template_skills rows created: N
- Unmatched JSONB tags: N (with list)
- persona_skills rows migrated to skill_id: N/17
- Unmatched persona_skills: N (with list)

Important: The script MUST import from `../src/db/client.js` for the pool connection (follow the pattern used by other scripts in the backend directory). Use `import { pool } from '../src/db/client.js';` and wrap in async main with proper exit.

Example structure:
```typescript
import { pool } from '../src/db/client.js';

async function main() {
  console.log('=== SOT Migration: Skills Source of Truth ===\n');

  // Step 1: DDL
  await pool.query('ALTER TABLE persona_skills ADD COLUMN IF NOT EXISTS skill_id TEXT');

  // Step 2: template_skills population
  const templates = (await pool.query('SELECT id, skills FROM agent_templates')).rows;
  const allSkills = (await pool.query('SELECT id FROM skills')).rows.map(r => r.id);
  const skillSet = new Set(allSkills);

  let tsCreated = 0;
  let unmatched: string[] = [];

  for (const t of templates) {
    const tags: string[] = typeof t.skills === 'string' ? JSON.parse(t.skills) : (t.skills ?? []);
    let sortIdx = 0;
    for (const tag of tags) {
      // Exact match
      if (skillSet.has(tag)) {
        await pool.query(
          'INSERT INTO template_skills (template_id, skill_id, sort_order) VALUES ($1, $2, $3) ON CONFLICT DO NOTHING',
          [t.id, tag, sortIdx++]
        );
        tsCreated++;
        continue;
      }
      // Fuzzy: find skills.id containing the tag
      const fuzzy = (await pool.query(
        "SELECT id FROM skills WHERE id LIKE '%' || $1 || '%' LIMIT 1",
        [tag]
      )).rows[0];
      if (fuzzy) {
        await pool.query(
          'INSERT INTO template_skills (template_id, skill_id, sort_order) VALUES ($1, $2, $3) ON CONFLICT DO NOTHING',
          [t.id, fuzzy.id, sortIdx++]
        );
        tsCreated++;
      } else {
        unmatched.push(`${t.id}: ${tag}`);
      }
    }
  }

  // Step 3: persona_skills migration
  const ps = (await pool.query('SELECT persona_id, skill_name FROM persona_skills WHERE skill_id IS NULL')).rows;
  let psMigrated = 0;
  let psUnmatched: string[] = [];
  for (const row of ps) {
    if (skillSet.has(row.skill_name)) {
      await pool.query('UPDATE persona_skills SET skill_id = $1 WHERE persona_id = $2 AND skill_name = $3',
        [row.skill_name, row.persona_id, row.skill_name]);
      psMigrated++;
      continue;
    }
    const fuzzy = (await pool.query(
      "SELECT id FROM skills WHERE id LIKE '%' || $1 || '%' LIMIT 1",
      [row.skill_name]
    )).rows[0];
    if (fuzzy) {
      await pool.query('UPDATE persona_skills SET skill_id = $1 WHERE persona_id = $2 AND skill_name = $3',
        [fuzzy.id, row.persona_id, row.skill_name]);
      psMigrated++;
    } else {
      psUnmatched.push(`${row.persona_id}: ${row.skill_name}`);
    }
  }

  // Step 4: Report
  console.log(`\n=== Migration Report ===`);
  console.log(`Templates processed: ${templates.length}`);
  console.log(`template_skills rows created: ${tsCreated}`);
  console.log(`Unmatched JSONB tags: ${unmatched.length}`);
  if (unmatched.length > 0) console.log(`  ${unmatched.slice(0, 20).join('\n  ')}${unmatched.length > 20 ? `\n  ... and ${unmatched.length - 20} more` : ''}`);
  console.log(`persona_skills migrated: ${psMigrated}/17`);
  if (psUnmatched.length > 0) console.log(`Unmatched persona_skills: ${psUnmatched.join(', ')}`);

  await pool.end();
  console.log('\nDone.');
}

main().catch(e => { console.error(e); process.exit(1); });
```
