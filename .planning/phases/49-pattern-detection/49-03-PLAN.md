<objective>
LRN-03 — Make project-level directive scoping a first-class, auto-derived part of `/context`. The schema already supports it (83 `scope='project'` rows in production, 64 for `ymc.capital`); the only missing piece is server-side auto-derivation when the caller passes only `cwd`. This plan wires `detectContext` (shipped by plan 49-04) into the /context handler so a curl with `cwd=/home/lobster/projects/ymc.capital` (no explicit `?project=`) returns the ymc.capital project directives alongside the software-silo section.

Purpose: Without this, every caller (CLI hook, future MSF callers, agent invocations) must explicitly compute and pass `?project=X`. With this, the server is the single source of truth for cwd→project mapping. Net effect: future code that only knows the cwd (e.g., dream-worker's per-turn `cwd` field — Phase 51 onwards) can attribute work to a project without re-implementing the regex.

Output: intellect.ts /context handler refactored to derive `effectiveProject` from `?project= ?? detectedContext.projectId`, project-directive and concept and episode queries all using `effectiveProject` symmetrically, project-directive section header annotated when server-derived. Plus a non-blocking SQL migration adding a forward-investment partial index on `(scope, scope_id, status) WHERE status='active'`. Zero behavior change for existing hook callers (they pass both query params, explicit wins).

```sql
-- Migration 049: directives (scope, scope_id, status) partial index
-- Phase 49 LRN-03 — forward-investment optimization for project-scope lookups
--
-- Rationale: /context handler queries `WHERE scope='project' AND scope_id=$1
-- AND status='active'` on every CLI session start. The existing index
-- `idx_directives_scope (scope, status)` is sufficient for today's 83 rows
-- but scales poorly as project-scope grows.
--
-- Partial index (WHERE status='active') keeps the index small — archived rows
-- are excluded. CONCURRENTLY would be safer for live tables but Postgres
-- migration runners typically run inside a transaction; CREATE INDEX IF NOT
-- EXISTS is non-blocking for small tables (directives ≈100 rows today).
--
-- Rollback (manual): DROP INDEX IF EXISTS idx_directives_scope_scope_id_status;

CREATE INDEX IF NOT EXISTS idx_directives_scope_scope_id_status
  ON directives (scope, scope_id, status)
  WHERE status = 'active';
```

Here's the concrete patch sketch — write the full handler from this:

```typescript
fastify.get('/context', async (request, reply) => {
  const { project, scope, cwd, session_id } = request.query as {
    project?: string;
    scope?: string;
    cwd?: string;
    session_id?: string;
  };
  void scope;

  // ── Phase 49 LRN-03/04: derive context once at the top ──
  // detectContext composes detectSilos + detectProject. Silos used by the
  // silo section below; projectId is the server-side cwd→project mapping.
  // Explicit ?project= wins (back-compat with the hook which passes both).
  let detectedContext: { silos: Array<{ id: string; displayName: string }>; projectId: string | null } = {
    silos: [],
    projectId: null,
  };
  try {
    detectedContext = await detectContext(
      { cwd, projectName: project, sessionId: session_id },
      pool,
    );
  } catch (err) {
    request.log.warn({ err }, '[intellect] detectContext failed — continuing without silo/project derivation');
  }
  const effectiveProject: string | null = project ?? detectedContext.projectId ?? null;
  const projectIdSource: 'query' | 'cwd' | 'none' =
    project ? 'query' : detectedContext.projectId ? 'cwd' : 'none';
  const projectIsServerDerived = projectIdSource === 'cwd';

  // System directives — unchanged
  const { rows: systemDirectives } = await pool.query<DirectiveRow>(
    `SELECT id, scope, scope_id, content, priority, verified_at
     FROM directives WHERE status = 'active' AND scope = 'workspace'
     ORDER BY priority DESC`,
  );

  // Project directives — now from effectiveProject (was: project)
  let projectDirectives: DirectiveRow[] = [];
  if (effectiveProject) {
    const { rows } = await pool.query<DirectiveRow>(
      `SELECT id, scope, scope_id, content, priority, verified_at
         FROM directives
        WHERE status = 'active' AND scope = 'project' AND scope_id = $1
        ORDER BY priority DESC`,
      [effectiveProject],
    );
    projectDirectives = rows;
  }

  // Concepts — now from effectiveProject (was: project)
  const conceptQuery = effectiveProject
    ? `SELECT id, scope, scope_id, content, trust_tier, confidence_score
       FROM concepts
       WHERE status = 'active' AND (scope = 'global' OR (scope = 'project' AND scope_id = $1))
       ORDER BY confidence_score DESC, last_used_at DESC NULLS LAST
       LIMIT 20`
    : `SELECT id, scope, scope_id, content, trust_tier, confidence_score
       FROM concepts
       WHERE status = 'active' AND scope = 'global'
       ORDER BY confidence_score DESC, last_used_at DESC NULLS LAST
       LIMIT 10`;
  const { rows: concepts } = await pool.query<ConceptRow>(
    conceptQuery,
    effectiveProject ? [effectiveProject] : [],
  );

  // Episodes — now from effectiveProject (was: project)
  let episodes: EpisodeRow[] = [];
  if (effectiveProject) {
    const { rows } = await pool.query<EpisodeRow>(
      `SELECT id, scope_id, summary, files_changed_json, created_at
       FROM episodes
       WHERE (scope = 'project' AND scope_id = $1) OR scope = 'workspace'
       ORDER BY created_at DESC LIMIT 5`,
      [effectiveProject],
    );
    episodes = rows;
  } else {
    const { rows } = await pool.query<EpisodeRow>(
      `SELECT id, scope_id, summary, files_changed_json, created_at
       FROM episodes ORDER BY created_at DESC LIMIT 5`,
    );
    episodes = rows;
  }

  // ... (skill-recs block unchanged; it does not depend on project)

  // Section assembly
  const sections: string[] = [];
  sections.push('## Porter Context');
  if (effectiveProject) sections.push(`Project: **${effectiveProject}**`);
  sections.push('');

  if (systemDirectives.length > 0) {
    sections.push('### System Directives');
    for (const d of systemDirectives.slice(0, 15)) sections.push(`- ${d.content}`);
    sections.push('');
  }

  // Silo section — use detectedContext.silos directly (no second detectSilos call)
  try {
    const silos = detectedContext.silos;
    if (silos.length > 0) {
      const siloIds = silos.map((s) => s.id);
      const siloDirectivesRes = await pool.query<{
        id: string; content: string; priority: number; scope_id: string;
      }>(
        `SELECT id, content, priority, scope_id FROM directives
         WHERE status = 'active' AND scope = 'silo' AND scope_id = ANY($1::text[])
         ORDER BY priority DESC LIMIT 20`,
        [siloIds],
      );
      const bySilo = new Map<string, string[]>();
      for (const row of siloDirectivesRes.rows) {
        const bucket = bySilo.get(row.scope_id) ?? [];
        bucket.push(`- ${row.content}`);
        bySilo.set(row.scope_id, bucket);
      }
      for (const silo of silos) {
        const lines = bySilo.get(silo.id);
        if (!lines || lines.length === 0) continue;
        sections.push(`## Silo: ${silo.displayName} — Operating Rules\n${lines.join('\n')}`);
        sections.push('');
      }
    }
  } catch (err) {
    request.log.warn({ err }, '[intellect] silo injection failed — continuing without silo section');
  }

  // Project Directives — header annotated when server-derived
  if (projectDirectives.length > 0 && effectiveProject) {
    const suffix = projectIsServerDerived ? ' — server-derived' : '';
    sections.push(`### Project Directives (${effectiveProject})${suffix}`);
    for (const d of projectDirectives) sections.push(`- ${d.content}`);
    sections.push('');
  }

  // ... (remaining sections — Recent Sessions, Concepts, Skills, Tools — unchanged)

  return reply.send(ok({
    context: sections.join('\n'),
    stats: {
      systemDirectives: systemDirectives.length,
      projectDirectives: projectDirectives.length,
      episodes: episodes.length,
      concepts: concepts.length,
      projectIdSource,                  // ← NEW (LRN-03 observability)
      effectiveProject,                  // ← NEW (LRN-03 observability)
    },
  }));
});
```
