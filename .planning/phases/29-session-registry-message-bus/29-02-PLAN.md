    ```typescript
      correlationId?: string;
      sourceAgent?: string;
      sourceGateway?: string;
      targetAgent?: string;
      targetGateway?: string;
      intent: string;
      payload: Record<string, unknown>;
      hopCount: number;
    }

    export interface MsgBusEventUpdate {
      status: 'delivered' | 'failed';
      dispatchLogId?: string;
      latencyMs?: number;
      responsePayload?: Record<string, unknown>;
    }
    ```

    logMsgBusEvent(init: MsgBusEventInit): Promise&lt;string&gt;
    - Generates id = crypto.randomUUID()
    - INSERT INTO msg_bus_events (id, correlation_id, source_agent, source_gateway, target_agent, target_gateway, intent, payload, hop_count, status, created_at)
    - status = 'pending'
    - Returns id

    updateMsgBusEvent(id: string, update: MsgBusEventUpdate): Promise&lt;void&gt;
    - UPDATE msg_bus_events SET status=$status, dispatch_log_id=$dispatchLogId, latency_ms=$latencyMs, response_payload=$responsePayload, delivered_at=EXTRACT(EPOCH FROM NOW()) WHERE id=$id
    - Only sets non-null fields — use conditional SQL or always pass all fields with null fallback.

    Both functions are non-blocking — callers .catch(() => {}) them if non-critical.
    Do NOT make them fire-and-forget internally — the route needs the msgBusId back from logMsgBusEvent.
  </action>
