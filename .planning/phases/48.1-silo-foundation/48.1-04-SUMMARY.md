## Decisions Made

- **Stdin parse uses a soft 1s timeout, not blocking:** the hook must not hang if Claude Code is misconfigured or pipes nothing. Resolve `null` on timeout, fall through to the legacy `project`-only call.
- **stderr (not stdout) for the warn line:** stdout content gets injected into the model's context — adding `[porter-session-start] stdin payload missing` there would pollute every fallback session. Stderr is captured by `journalctl --user -u porter-fastify` (or wherever Claude Code forwards hook stderr) and stays out of the model's view.
- **`URLSearchParams` over manual encoding:** safer for edge characters in `cwd` (e.g., spaces in unusual project paths) and reads cleaner.
- **Hook NOT committed — explicit precedent from Plan 03:** the global Claude Code hook directory is the user's environment, not the Porter repo. Verbatim reproduction in this SUMMARY is the re-deployment artifact (same pattern Plan 03 used for `porter-user-prompt.js`).
- **Version bump touched 4 real locations, not the textbook 8:** the 8-location rule in Porter/CLAUDE.md is aspirational — empirical grep showed only 4 hardcoded `v6.11.0` strings (3 backend + 1 CHANGELOG). The remaining 4 logical surfaces (admin frontend badge, admin page footer, SSE version, /api/version) either share a runtime source (admin reads `/health`) or don't exist as hardcoded strings in current architecture. Documented above under BLOCKER 4.
- **`systemctl stop && start` not `restart`:** Plans 01 + 02 both observed `restart` letting the npm exec tsx wrapper absorb SIGTERM and re-exec from a cached PID, masking source changes. Explicit stop+start produces a fresh PID (252036 here) with clean module reload.

```javascript
#!/usr/bin/env node
// Porter Session Start Hook
// Queries Porter Intellect API for scoped memory context and injects it.
// Detects project from working directory and loads relevant memory on-demand.
//
// Phase 48.1 Plan 04: now reads the SessionStart event payload from stdin
// (JSON with session_id, hook_event_name, cwd, …), extracts session_id + cwd,
// and forwards both to /api/v1/intellect/context as query params so the silo
// detector (Phase 48.1 Plan 02) can see them. Falls back gracefully to the
// project-only call path if stdin is missing/unparseable.

const http = require('http');
const path = require('path');
const fs = require('fs');

const PORTER_HOST = '127.0.0.1';
const PORTER_PORT = 3001;

// ── Project detection ─────────────────────────────────────────────────────

function detectProject(cwd) {
  const effectiveCwd = cwd || process.cwd();
  // /home/lobster/projects/<name>/ or subdirs
  const match = effectiveCwd.match(/^\/home\/lobster\/projects\/([^/]+)/);
  if (match) return match[1];
  return null;
}

// ── Stdin event payload reader (SessionStart hook protocol) ───────────────
// Claude Code pipes a JSON event payload on stdin. We read it with a short
// timeout so the hook never hangs if Claude doesn't pipe anything.
function readStdinJson(timeoutMs = 1000) {
  return new Promise((resolve) => {
    let buf = '';
    const t = setTimeout(() => resolve(null), timeoutMs);
    try {
      process.stdin.setEncoding('utf8');
    } catch {
      // setEncoding can throw if stdin is already closed
      clearTimeout(t);
      return resolve(null);
    }
    process.stdin.on('data', (chunk) => { buf += chunk; });
    process.stdin.on('end', () => {
      clearTimeout(t);
      if (!buf || !buf.trim()) return resolve(null);
      try { resolve(JSON.parse(buf)); } catch { resolve(null); }
    });
    process.stdin.on('error', () => { clearTimeout(t); resolve(null); });
  });
}

// ── Porter API query ──────────────────────────────────────────────────────

function queryPorterContext(project, cwd, sessionId) {
  return new Promise((resolve) => {
    const params = new URLSearchParams();
    if (project)   params.set('project', project);
    if (cwd)       params.set('cwd', cwd);
    if (sessionId) params.set('session_id', sessionId);
    const qs = params.toString() ? `?${params.toString()}` : '';
    const req = http.request(
      {
        hostname: PORTER_HOST,
        port: PORTER_PORT,
        path: `/api/v1/intellect/context${qs}`,
        method: 'GET',
        timeout: 3000,
      },
      (res) => {
        let body = '';
        res.on('data', (chunk) => (body += chunk));
        res.on('end', () => {
          try {
            const parsed = JSON.parse(body);
            resolve(parsed.data || null);
          } catch {
            resolve(null);
          }
        });
      }
    );
    req.on('error', () => resolve(null));
    req.on('timeout', () => {
      req.destroy();
      resolve(null);
    });
    req.end();
  });
}

// ── Checkpoint reader (project-specific) ──────────────────────────────────

function readProjectCheckpoint(project) {
  if (!project) return null;
  const candidates = [
    `/home/lobster/projects/${project}/CHECKPOINT.md`,
    `/home/lobster/projects/${project}/tasks/checkpoint.md`,
    `/home/lobster/projects/${project}/admin/tasks/checkpoint.md`,
  ];
  for (const p of candidates) {
    try {
      return fs.readFileSync(p, 'utf8').slice(0, 1500);
    } catch {}
  }
  return null;
}

// ── Git log reader (project-specific) ─────────────────────────────────────

function readGitLog(project) {
  if (!project) return null;
  const { execSync } = require('child_process');
  const projectDir = `/home/lobster/projects/${project}`;
  try {
    return execSync('git log --oneline -5 --format="%h %s (%ar)"', {
      cwd: projectDir,
      encoding: 'utf8',
      timeout: 3000,
    }).trim();
  } catch {
    return null;
  }
}

// ── Main ──────────────────────────────────────────────────────────────────

async function main() {
  // Read SessionStart event payload from stdin (Phase 48.1 Plan 04)
  const eventData = await readStdinJson();
  if (eventData === null) {
    try { process.stderr.write('[porter-session-start] stdin payload missing or unparseable — falling back to project-only call\n'); } catch {}
  }

  const sessionId = (eventData && eventData.session_id) ? String(eventData.session_id) : null;
  const eventCwd  = (eventData && eventData.cwd)        ? String(eventData.cwd)        : process.cwd();
  const project = detectProject(eventCwd);
  const lines = [];

  // Query Porter for scoped memory context (now with cwd + session_id so the
  // silo detector can pick the right silo)
  const intellectContext = await queryPorterContext(project, eventCwd, sessionId);

  if (intellectContext && intellectContext.context) {
    lines.push(intellectContext.context);
    lines.push('');
  } else {
    // Fallback: minimal header
    lines.push('## Porter Session Context');
    lines.push('');
    if (project) {
      lines.push(`Working on project: **${project}**`);
    } else {
      lines.push('No project detected. Ask Moe which project to work on.');
    }
    lines.push('');
  }

  // Read project checkpoint if available
  if (project) {
    const checkpoint = readProjectCheckpoint(project);
    if (checkpoint) {
      lines.push('### Checkpoint (truncated)');
      lines.push(checkpoint);
      lines.push('');
    }

    const gitLog = readGitLog(project);
    if (gitLog) {
      lines.push(`### Recent commits (${project})`);
      lines.push(gitLog);
      lines.push('');
    }
  }

  // Output for Claude Code to inject
  process.stdout.write(lines.join('\n'));
}

main().catch(() => {
  // Never fail the hook — just print minimal fallback
  process.stdout.write('## Porter Session Context\n\n(Intellect unavailable)\n');
});
```
