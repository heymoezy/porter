           LEFT JOIN skills s ON s.id = sep.skill_id

    1. **Inline SKILLS.md regenerator** — Add a helper function inside the route plugin (NOT imported from backend):
       ```typescript
       async function regenSkillsManifest(personaId: string): Promise<void> {
         // Look up persona name
         const persona = await queryOne<{ name: string }>(
           'SELECT name FROM personas WHERE id = $1', [personaId]
         );
         if (!persona) return; // persona deleted, skip

         // Gather assigned skills
         const rows = await queryAll<{
           skill_id: string; enabled: number; name: string; description: string; category: string;
         }>(`
           SELECT ps.skill_id, ps.enabled, s.name, s.description, s.category
           FROM persona_skills ps
           JOIN skills s ON s.id = ps.skill_id
           WHERE ps.persona_id = $1 AND ps.skill_id IS NOT NULL
           ORDER BY s.category, s.name
         `, [personaId]);

         // Generate markdown (same format as backend/src/services/skills-manifest.ts)
         const now = new Date().toISOString();
         const active = rows.filter(r => r.enabled === 1 || r.enabled === true as any);
         const disabled = rows.filter(r => r.enabled === 0 || r.enabled === false as any);

         let md = `# SKILLS -- ${persona.name}\n\n`;
         md += `> Auto-generated manifest. Do not edit manually.\n`;
         md += `> Source: persona_skills table | Generated: ${now}\n\n`;

         if (active.length === 0) {
           md += `## Active Skills (0)\n\n_(no skills assigned)_\n`;
         } else {
           md += `## Active Skills (${active.length})\n\n`;
           const byCategory = new Map<string, typeof active>();
           for (const r of active) {
             const cat = r.category || 'uncategorized';
             if (!byCategory.has(cat)) byCategory.set(cat, []);
             byCategory.get(cat)!.push(r);
           }
           for (const [cat, skills] of byCategory) {
             md += `### ${cat}\n\n`;
             for (const s of skills) {
               md += `- **${s.skill_id}** -- ${s.name}${s.description ? `: ${s.description.slice(0, 120)}` : ''}\n`;
               md += `  Pack: \`skills/${s.skill_id}/\`\n`;
             }
             md += `\n`;
           }
         }
         if (disabled.length > 0) {
           md += `## Disabled Skills (${disabled.length})\n\n`;
           for (const s of disabled) md += `- ~~${s.skill_id}~~ -- ${s.name}\n`;
           md += `\n`;
         }
         md += `_Last regenerated: ${now}_\n`;

         // Write to disk — check both possible persona directories
         const fs = await import('fs/promises');
         const path = await import('path');
         const homeDir = process.env.HOME || '/home/lobster';
         // Use projects/porter/personas (where personas actually live)
         const personaDir = path.join(homeDir, 'projects/porter/personas', personaId);
         await fs.mkdir(personaDir, { recursive: true });
         await fs.writeFile(path.join(personaDir, 'SKILLS.md'), md, 'utf-8');
       }
       ```

    2. **POST /proposals/:proposalId/approve**:
       - Fetch proposal WHERE id=$1 AND status='pending'. Return 404 if not found or already reviewed.
       - Get current effectiveness_before from persona_skills for this persona+skill.
       - Apply change based on change_type:
         - `remove_skill`: DELETE FROM persona_skills WHERE persona_id=$1 AND (skill_id=$2 OR skill_name=$2)
         - `add_skill`: INSERT INTO persona_skills (persona_id, skill_id, skill_name, enabled) VALUES ($1, $2, $2, 1) ON CONFLICT DO NOTHING
         - `rewrite_prompt`: This is a proposal for manual follow-up. Update proposed_change JSONB to include "action_needed: rewrite prompt.md for skill pack". No auto-mutation on persona_skills for this type.
         - `enrich_examples`: Same as rewrite_prompt — proposal flags need for manual enrichment.
       - Call `regenSkillsManifest(proposal.persona_id)`.
       - INSERT into skill_evolution_events with all fields including effectiveness_before.
       - UPDATE proposal: status='approved', reviewed_at=EXTRACT(EPOCH FROM NOW()), reviewed_by=username from req.user.
       - Return `ok({ approved: true, proposal_id: proposalId })`.
