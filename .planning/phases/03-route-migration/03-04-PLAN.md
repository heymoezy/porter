    **Step 4: Update frontend/src/main.tsx with React Router**

    Replace the current direct App render with React Router:
    ```typescript
    import { StrictMode } from 'react';
    import { createRoot } from 'react-dom/client';
    import { createBrowserRouter, RouterProvider } from 'react-router-dom';
    import './index.css';
    import './design-system/index.css';
    import App from './App.tsx';
    import { LoginPage } from './pages/LoginPage.tsx';
    import { RegisterPage } from './pages/RegisterPage.tsx';

    const router = createBrowserRouter([
      { path: '/login', element: <LoginPage /> },
      { path: '/register', element: <RegisterPage /> },
      { path: '/*', element: <App /> },
    ], {
      basename: '/v2',
    });

    createRoot(document.getElementById('root')!).render(
      <StrictMode>
        <RouterProvider router={router} />
      </StrictMode>,
    );
    ```

    IMPORTANT: The basename must be '/v2' because Vite's base path is '/v2/' (see vite.config.ts). React Router needs to know its routes are relative to /v2/. So the login page is at /v2/login, register at /v2/register, and the main app at /v2/.
