```python
def _migrate_projects_from_json():
    """One-shot migration: move projects from porter_config.json to SQLite.
    Guarded by schema_migrations entry to prevent re-run."""
    migration_id = "migrate_projects_from_json_v1"
    conn = _db_conn()
    try:
        # Check if already migrated
        row = conn.execute(
            "SELECT id FROM schema_migrations WHERE id = ?", (migration_id,)
        ).fetchone()
        if row:
            return  # Already migrated

        # Read projects from config
        config_projects = _config.get("projects", {})
        if not config_projects:
            # Mark as migrated even if nothing to migrate
            conn.execute(
                "INSERT INTO schema_migrations (id) VALUES (?)", (migration_id,)
            )
            conn.commit()
            mlog.emit("info", "db", "migration.skip",
                      "No projects in config to migrate")
            return

        # Insert each project
        count = 0
        for pid, proj in config_projects.items():
            import json as _json
            conn.execute("""
                INSERT OR IGNORE INTO projects
                    (id, name, slug, type, status, description, owner_id,
                     milestones, artifacts, links, metadata, created_at, updated_at)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """, (
                proj.get("id", pid),
                proj.get("name", "Untitled"),
                proj.get("slug") or proj.get("name", "untitled").lower().replace(" ", "-"),
                proj.get("type", "custom"),
                proj.get("status", "active"),
                proj.get("description", ""),
                proj.get("owner_id") or proj.get("created_by", "moe"),
                _json.dumps(proj.get("milestones", [])),
                _json.dumps(proj.get("artifacts", [])),
                _json.dumps(proj.get("links", [])),
                _json.dumps({k: v for k, v in proj.items()
                            if k not in ("id", "name", "slug", "type", "status",
                                        "description", "owner_id", "milestones",
                                        "artifacts", "links", "created_at", "updated_at")}),
                proj.get("created_at") or __import__('time').time(),
                proj.get("updated_at") or __import__('time').time(),
            ))
            count += 1

        # Record migration
        conn.execute(
            "INSERT INTO schema_migrations (id) VALUES (?)", (migration_id,)
        )
        conn.commit()

        # Remove projects from config and save
        if "projects" in _config:
            del _config["projects"]
            _save_config()

        mlog.emit("info", "db", "migration.complete",
                  f"Migrated {count} projects from JSON to SQLite")

    except Exception as e:
        mlog.emit("error", "db", "migration.failed",
                  f"Projects migration failed: {e}",
                  extra={"exc_type": type(e).__name__})
        raise
```

```python
    """Load a project by ID from SQLite."""
    if not project_id:

    conn = _db_conn()
```

```python
def _project_list(owner_id=None, status=None):
    """List projects from SQLite, optionally filtered by owner or status."""
    conn = _db_conn()
    try:
        query = "SELECT * FROM projects"
        params = []
        conditions = []
        if owner_id:
            conditions.append("owner_id = ?")
            params.append(owner_id)
        if status:
            conditions.append("status = ?")
            params.append(status)
        if conditions:
            query += " WHERE " + " AND ".join(conditions)
        query += " ORDER BY updated_at DESC"
        rows = conn.execute(query, params).fetchall()
        return [dict(r) for r in rows]
    except Exception as e:
        mlog.emit("warn", "db", "project.list_failed", str(e))
        return []
```
