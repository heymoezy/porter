```python
#!/usr/bin/env python3
"""Patch porter.py to add mlog.emit() to three silent exception handlers."""
import pathlib

porter = pathlib.Path("/home/lobster/documents/porter/porter.py")
content = porter.read_text()

# Change 1: stale connection close (inside _db_conn try/except)
# Context: this is inside a nested try block within the stale-check
old1 = '''            except Exception:\n                pass\n            _thread_local.conn = None'''
new1 = '''            except Exception as _e:\n                mlog.emit("debug", "db", "conn.stale_close_failed",\n                          f"Stale connection close failed: {_e}",\n                          extra={"exc_type": type(_e).__name__})\n            _thread_local.conn = None'''
assert old1 in content, f"Change 1 target not found"
content = content.replace(old1, new1, 1)

# Change 2: unified rate-limit ValueError

new2 = '''                except ValueError as _e:\n                    mlog.emit("debug", "ai", "ratelimit.parse_failed",\n                              f"Unified rate-limit header parse failed: {_e}",\n                              extra={"exc_type": "ValueError"})'''
assert old2 in content, f"Change 2 target not found"
content = content.replace(old2, new2, 1)

# Change 3: fallback rate-limit ValueError/ZeroDivisionError
old3 = '''            except (ValueError, ZeroDivisionError):\n                pass'''
new3 = '''            except (ValueError, ZeroDivisionError) as _e:\n                mlog.emit("debug", "ai", "ratelimit.parse_failed",\n                          f"Fallback rate-limit header parse failed: {_e}",\n                          extra={"exc_type": type(_e).__name__})'''
assert old3 in content, f"Change 3 target not found"
content = content.replace(old3, new3, 1)

porter.write_text(content)
print("All 3 silent exception handlers patched with mlog.emit()")
```
