    """Populate DEFAULT_MOUNTS from environment or config."""
    data_dir = os.environ.get("PORTER_DATA_DIR", "")
    if data_dir:
        base = Path(data_dir)
        if base.exists():
            DEFAULT_MOUNTS["data"] = base
    # Only add directories that actually exist
    for name, env_key in [("documents", "PORTER_DOCUMENTS_DIR"),
                           ("uploads", "PORTER_UPLOADS_DIR"),
                           ("websites", "PORTER_WEBSITES_DIR")]:
        d = os.environ.get(env_key, "")
        if d and Path(d).exists():
            DEFAULT_MOUNTS[name] = Path(d)

**5. Implement _boot_sequence():**
```python
def _boot_sequence():
    """Detect -> notify -> install -> configure -> verify -> badge.
    Runs once at startup before accepting HTTP requests."""
    results = {}

    # 1. Required: Python stdlib (always available)
    results['python'] = {'ok': True, 'version': sys.version.split()[0]}

    # 2. Required: SQLite DB writable
    try:
        conn = _db_conn()
        conn.execute("SELECT 1")
        results['sqlite'] = {'ok': True, 'path': str(DB_PATH)}
    except Exception as e:
        results['sqlite'] = {'ok': False, 'error': str(e)}

    # 3. Required: Data directory exists and is writable
    _DATA_DIR.mkdir(parents=True, exist_ok=True)
    results['data_dir'] = {'ok': _DATA_DIR.exists() and os.access(str(_DATA_DIR), os.W_OK),
                           'path': str(_DATA_DIR)}

    # 4. Optional: Node.js (for Fastify backend)
    try:
        import subprocess
        node_v = subprocess.run(['node', '--version'], capture_output=True, text=True, timeout=5)
        results['node'] = {'ok': node_v.returncode == 0, 'version': node_v.stdout.strip()}
    except Exception:
        results['node'] = {'ok': False, 'error': 'node not found'}

    # 5. Optional: Ollama (local models)
    try:
        import urllib.request
        req = urllib.request.urlopen('http://127.0.0.1:11434/api/tags', timeout=3)
        results['ollama'] = {'ok': True}
    except Exception:
        results['ollama'] = {'ok': False, 'error': 'Ollama not reachable'}

    # 6. Optional: OpenClaw gateway (skip check if OPENCLAW_URL not set)
    openclaw_url = os.environ.get('OPENCLAW_URL', '')
    if openclaw_url:
        try:
            import urllib.request
            req = urllib.request.urlopen(f'{openclaw_url}/health', timeout=3)
            results['openclaw'] = {'ok': True, 'url': openclaw_url}
        except Exception:
            results['openclaw'] = {'ok': False, 'error': f'OpenClaw not reachable at {openclaw_url}'}
    else:
        results['openclaw'] = {'ok': False, 'error': 'OPENCLAW_URL not configured (skipped)'}

    # Store for UI badging
    global _capabilities
    _capabilities = results

    # Log summary
    required_ok = all(results.get(k, {}).get('ok', False) for k in ['python', 'sqlite', 'data_dir'])
    optional_missing = [k for k in ['node', 'ollama', 'openclaw'] if not results.get(k, {}).get('ok', False)]

    if required_ok and not optional_missing:
        mlog.emit("info", "system", "boot.ok", "All capabilities verified",
                  extra={"capabilities": results})
    elif required_ok:
        mlog.emit("warn", "system", "boot.degraded",
                  f"Optional capabilities missing: {optional_missing}",
                  extra={"capabilities": results})
    else:
        failed = [k for k in ['python', 'sqlite', 'data_dir'] if not results.get(k, {}).get('ok', False)]
        mlog.emit("error", "system", "boot.critical",
                  f"Required capabilities failed: {failed}",
                  extra={"capabilities": results})

    return results
```
