**Zod schemas at top:**
```typescript
const invitationSchema = z.object({
  email: z.string().email(),
  role: z.enum(['view', 'chat', 'edit', 'admin']),  // NOT 'owner' — owner cannot be assigned
});
const batchInviteSchema = z.object({
  invitations: z.array(invitationSchema).min(1).max(50),
});
const changeRoleSchema = z.object({
  role: z.enum(['view', 'chat', 'edit', 'admin']),
});
const acceptInviteSchema = z.object({
  token: z.string().min(32),
  password: z.string().min(8).optional(),
  display_name: z.string().min(1).max(100).optional(),
});
```

**Route 1: POST /api/v1/projects/:id/collaborators (batch invite)**
- preHandler: `[fastify.requireAuth, fastify.requireProjectAccess('edit')]`
- Validate body with `batchInviteSchema`
- For each invitation in the batch:
  a. Check if collaborator record already exists for this (project_id, email):
     - If `status='active'`: add to `failed` list with reason `'already_active'`
     - If `status='revoked'`: re-invite by setting `status='active'`, `role` to new role, `invite_token=null`, `revoked_at=null`, `revoked_by=null`; log `'reinstated'` event; add to `succeeded`
     - If `status='pending'`: update role if changed; add to `failed` with reason `'already_pending'`
  b. If no record exists:
     - Check if user exists in `users` table by email. If found, set `username` on the collaborator record.
     - If user does NOT exist: create a pending user: `INSERT INTO users (username, email, password_hash, salt, role, status) VALUES (email_slug, email, '', '', 'operator', 'pending')`. Use the email local part (before @) + random 4 chars as username. Use `INSERT OR IGNORE` to handle races.
     - Generate token: `crypto.randomBytes(32).toString('hex')`
     - Insert into `project_collaborators`: id=`crypto.randomUUID()`, project_id, username (if known), email, role, status='pending', invite_token, invited_by=`request.sessionUser.username`
     - Log `'invited'` event in `collaboration_events`
     - Call `sendInviteEmail()` (imported from transactional-email.ts) with project name, inviter name, token, role
     - Schedule first drip reminder via `scheduleDripReminder()` (imported from scheduler helper — see Task 2)
     - Add to `succeeded` list
  c. Use `sqlite.transaction(() => { ... })()` to wrap each individual invite (not the whole batch) for atomicity
- Return: `ok({ succeeded: [...], failed: [...], total: invitations.length })`

**Route 2: GET /api/v1/projects/:id/collaborators (list)**
- preHandler: `[fastify.requireAuth, fastify.requireProjectAccess('view')]`
- Query: `SELECT pc.*, u.display_name, u.full_name FROM project_collaborators pc LEFT JOIN users u ON u.username = pc.username WHERE pc.project_id = ?`
- Return all records (pending, active, revoked) with user display info
- Return: `ok({ collaborators: [...], count })`

**Route 3: PATCH /api/v1/projects/:id/collaborators/:collab_id (change role)**
- preHandler: `[fastify.requireAuth, fastify.requireProjectAccess('admin')]`
- Validate body with `changeRoleSchema`
- Look up collaborator by id and project_id
- If not found: 404 `COLLABORATOR_NOT_FOUND`
- If target is owner role: 400 `CANNOT_MODIFY_OWNER` (owner role is immutable)
- If caller is admin (not owner) and target is also admin: 403 `FORBIDDEN` (admin cannot modify peer admins)
- Update `role`, `updated_at`
- Log `'role_changed'` event with `previous_role` and `new_role`
- Return: `ok({ collaborator: updated_record })`

**Route 4: DELETE /api/v1/projects/:id/collaborators/:collab_id (revoke)**
- preHandler: `[fastify.requireAuth, fastify.requireProjectAccess('admin')]`
- Look up collaborator by id and project_id
- If not found: 404 `COLLABORATOR_NOT_FOUND`
- If target role is `'owner'`: 400 `CANNOT_REVOKE_OWNER`
- If caller is admin (not owner) and target is also admin: 403 `FORBIDDEN`
- Set `status='revoked'`, `revoked_at=Date.now()/1000`, `revoked_by=request.sessionUser.username`
- Cancel pending drip jobs: `UPDATE agent_jobs SET status='cancelled', completed_at=unixepoch('now') WHERE trigger_type='invite_drip' AND json_extract(trigger_data, '$.collaborator_id') = ? AND status='pending'`
- Log `'revoked'` event
- Return: `ok({ revoked: true })`

**Route 5: POST /api/v1/collaborators/accept (accept invite — NO auth required)**
- This route is registered at a DIFFERENT prefix — NOT under `/projects/:id/collaborators`. It is at `/api/v1/collaborators/accept`.
- NO preHandler (no auth — token is the auth)
- Validate body with `acceptInviteSchema`
- Look up collaborator by `invite_token` where `status='pending'`
- If not found: 404 `INVALID_INVITE_TOKEN`
- If user does not exist by email: create user with `username` = email slug + random chars, `password_hash` and `salt` from the provided password (hash with crypto.pbkdf2Sync, 100000 iterations, 64 bytes, sha512 — match existing pattern in auth routes), `status='active'`, `email_verified=1`
- If user exists but `status='pending'`: update password_hash/salt if password provided, set `status='active'`
- Update collaborator: `status='active'`, `username=user.username`, `accepted_at=Date.now()/1000`, `invite_token=NULL`
- Cancel pending drip jobs for this collaborator
- Log `'accepted'` event
- Return: `ok({ project_id, role, username })`

```typescript
export function scheduleDripReminder(collaboratorId: string, dripCount: number): void {
  // Cadence: days 1-3 (daily), then weeks 1-4 (weekly), then monthly
  const offsetDays = dripCount < 3 ? 1 : dripCount < 7 ? 7 : 30;
  const scheduledFor = Date.now() / 1000 + offsetDays * 86400;

  sqlite.prepare(`
    INSERT INTO agent_jobs (id, agent_id, trigger_type, trigger_data, status, scheduled_for, created_at)
    VALUES (?, 'system', 'invite_drip', ?, 'pending', ?, unixepoch('now'))
  `).run(crypto.randomUUID(), JSON.stringify({ collaborator_id: collaboratorId }), scheduledFor);
}
```

if (job.trigger_type === 'invite_drip') {
  const data = JSON.parse(job.trigger_data || '{}');
  const collaboratorId = data.collaborator_id;
  if (!collaboratorId) throw new Error('Missing collaborator_id in drip job');

  // Load collaborator record
  const collab = sqlite.prepare(
    `SELECT pc.*, p.name as project_name, u2.display_name as inviter_name
     FROM project_collaborators pc
     LEFT JOIN projects p ON p.id = pc.project_id
     LEFT JOIN users u2 ON u2.username = pc.invited_by
     WHERE pc.id = ?`
  ).get(collaboratorId) as any;

  if (!collab || collab.status !== 'pending') {
    // Already accepted or revoked — no more drips
    return;
  }

  if (collab.drip_count >= MAX_DRIP_COUNT) {
    // Max drips reached — stop
    return;
  }

  // Send drip email
  const { sendDripReminder: sendDrip } = await import('./transactional-email.js');
  await sendDrip({

    projectName: collab.project_name || 'a project',
    inviterName: collab.inviter_name || collab.invited_by,
    role: collab.role,
    token: collab.invite_token,
    dripCount: collab.drip_count,
  });

  // Update drip tracking
  sqlite.prepare(
    `UPDATE project_collaborators SET drip_count = drip_count + 1, last_drip_at = unixepoch('now') WHERE id = ?`
  ).run(collaboratorId);

  // Log drip event
  sqlite.prepare(
    `INSERT INTO collaboration_events (project_id, collaborator_id, actor_username, event_type, detail, created_at)
     VALUES (?, ?, 'system', 'drip_sent', ?, unixepoch('now'))`

  // Schedule next drip (if under max)

    scheduleDripReminder(collaboratorId, collab.drip_count + 1);
  }
}
