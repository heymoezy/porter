- This can be done as a post-filter on the results, or by joining with project_collaborators:

```typescript
// In GET /sessions, update the non-admin query to include project access:
const query = `
  SELECT c.id, c.title, c.model_id, c.updated_at, c.username,
         c.project_id, c.metadata,
         (SELECT count(*) FROM chat_messages WHERE chat_id = c.id) as msg_count
  FROM chats c
  WHERE (c.username = @username OR c.username IS NULL)
```

**2. Add project-scoped access check to the POST /stream route:**

Currently the stream route only requires `requireAuth`. For project-context messages, it should verify the user has `'chat'` role on the project.

After the existing body parsing and message validation, add:
```typescript
// If a project context is provided, verify project access
const projectId = (body as any)?.project_id;
if (projectId) {
  if (request.sessionUser!.role !== 'platform_admin') {
    const projectAccess = sqlite.prepare(
      `SELECT role FROM project_collaborators WHERE project_id = ? AND username = ? AND status = 'active'`
    ).get(projectId, request.sessionUser!.username) as { role: string } | undefined;

    if (!projectAccess) {
      return reply.code(403).send(err('FORBIDDEN', 'No access to this project', request.id));
    }

    const roleOrder = ['view', 'chat', 'edit', 'admin', 'owner'];
    if (roleOrder.indexOf(projectAccess.role) < roleOrder.indexOf('chat')) {
      return reply.code(403).send(err('FORBIDDEN', 'Chat access required', request.id));
    }

    request.projectRole = projectAccess.role as ProjectRole;
  } else {
    request.projectRole = 'owner';
  }
}
```
