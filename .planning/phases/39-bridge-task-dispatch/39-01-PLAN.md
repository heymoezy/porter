```typescript
// ── Task Dispatch types (Phase 39) ──────────────────────────────────────────

export interface TaskRequest {
  prompt: string;            // The task description
  cwd: string;               // Working directory for the subprocess
  timeoutMs?: number;        // Default: 300_000 (5 min), max: 600_000 (10 min)
  tools?: string[];          // Optional tool allowlist (claude: --allowedTools)
}

export interface TaskEvent {
  type: 'progress' | 'result' | 'error' | 'tool_use' | 'tool_result';
  text?: string;             // Human-readable output chunk
  tool?: string;             // Tool name (for tool_use events)
  input?: unknown;           // Tool input (for tool_use events)
  exitCode?: number;         // Final exit code (for result/error)
  durationMs?: number;       // Total wall time (for result/error)
}

export interface TaskDispatchResult {
  taskId: string;
  gatewayType: string;
  model: string;
  status: 'complete' | 'failed' | 'cancelled';
  output: string;            // Full concatenated text output (max 1MB)
  durationMs: number;
  exitCode: number | null;
}

export type TaskStatus = 'queued' | 'running' | 'complete' | 'failed' | 'cancelled';
```

   - claude_cli: args = ['-p', '--dangerously-skip-permissions', '--output-format', 'stream-json', '--verbose', '--include-partial-messages', '--no-session-persistence', '--bare']. stdinPrompt = prompt. spawnCwd = cwd.

   - Throw for any other gateway type.

5. Export async generator function:
   async function* executeTask(binaryPath: string, gatewayType: GatewayType, prompt: string, cwd: string, signal: AbortSignal, timeoutMs?: number): AsyncGenerator<TaskEvent>

   Implementation:
   - Call validateCwd(cwd)
   - Call buildTaskArgs to get args, stdinPrompt, spawnCwd
   - Clamp timeout between DEFAULT_TIMEOUT_MS and MAX_TIMEOUT_MS
   - spawn(binaryPath, args, { cwd: spawnCwd, stdio: ['pipe', 'pipe', 'pipe'], env: { ...process.env, PORTER_BRIDGE_DISPATCH: '1' } })
   - child.stderr: collect into stderrChunks array (drain to prevent deadlock)
   - AbortSignal handler: SIGTERM then setTimeout 5s SIGKILL
   - Hard timeout: SIGTERM then setTimeout 5s SIGKILL
   - If stdinPrompt: child.stdin.write(stdinPrompt, 'utf8'); child.stdin.end()
   - Else: child.stdin.end()
   - readline loop on child.stdout:
     - Parse each line as JSON (skip non-JSON)
     - For Claude: extract text from type=assistant message.content[].text (accumulator delta pattern) and type=stream_event content_block_delta. Yield progress events. Detect tool_use from content blocks.
     - For Gemini: extract text from type=message role=assistant content. Yield progress events.
     - For Codex: extract text from type=item.completed item.type=message content[].output_text. Filter bwrap warnings from stderr.
     - Track total output size; if > MAX_OUTPUT_BYTES, truncate and yield a progress event with "[output truncated at 1MB]"
   - After readline loop ends, wait for child.close event
   - Yield final TaskEvent with type='result', exitCode from child exit code, durationMs

6. Export TASK_CAPABLE_TYPES, validateCwd, buildTaskArgs, getTaskQueue, executeTask
