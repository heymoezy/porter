         ```typescript
         export interface GatewayConfidence {
           gatewayType: string;
           avgScore: number;       // weighted average of outcome_score (1.0-5.0)
           totalRated: number;     // count of rated dispatches
           confidence: number;     // 0.0-1.0 — how much to trust this score (increases with totalRated)
           recentTrend: 'improving' | 'declining' | 'stable' | 'unknown';
           lastUpdated: number;    // epoch seconds
         }
         ```

       - In-memory cache: `Map<string, GatewayConfidence>` with TTL of 5 minutes
       - `refreshConfidence(): Promise<void>`:
         ```sql
         SELECT
           gateway_type,
           AVG(outcome_score) AS avg_score,
           COUNT(*) AS total_rated,
           AVG(CASE WHEN created_at > EXTRACT(EPOCH FROM NOW()) - 86400 THEN outcome_score END) AS recent_avg,
           AVG(CASE WHEN created_at <= EXTRACT(EPOCH FROM NOW()) - 86400 THEN outcome_score END) AS older_avg
         FROM bridge_dispatch_log
         WHERE outcome_score IS NOT NULL
         GROUP BY gateway_type
         ```
         - Compute confidence: `Math.min(1.0, totalRated / 50)` — reaches 1.0 after 50 rated dispatches
         - Compute recentTrend: if recent_avg > older_avg + 0.3 = 'improving', if < older_avg - 0.3 = 'declining', else 'stable', if no older data = 'unknown'
         - Store in cache with current timestamp

       - `getGatewayConfidence(gatewayType: string): Promise<GatewayConfidence | null>`:
         - If cache is stale (>5 min), call refreshConfidence first
         - Return cached entry or null if no data for that gateway

       - `getAllConfidenceScores(): Promise<GatewayConfidence[]>`:
         - If cache is stale, refresh first

    2. Create `backend/src/routes/v1/dispatch-outcome.ts`:
       ```typescript
       import type { FastifyInstance } from 'fastify';
       import { pool } from '../../db/client.js';
       import { refreshConfidence } from '../../services/bridge/routing-confidence.js';

       export async function dispatchOutcomeRoutes(app: FastifyInstance) {
         // SIN-03: Score a dispatch outcome
         app.post('/dispatches/:id/outcome', async (req, reply) => {
           const { id } = req.params as { id: string };
           const { score, note } = req.body as { score: number; note?: string };

           // Validate score: integer 1-5
           if (!score || score < 1 || score > 5 || !Number.isInteger(score)) {
             return reply.status(400).send({
               error: 'invalid_score',
               message: 'Score must be an integer from 1 to 5',
             });
           }

           // Verify dispatch exists
           const { rows } = await pool.query(
             'SELECT id FROM bridge_dispatch_log WHERE id = $1',
             [id]
           );
           if (rows.length === 0) {
             return reply.status(404).send({
               error: 'dispatch_not_found',
               message: `Dispatch ${id} not found`,
             });
           }

           // Update outcome
           await pool.query(
             `UPDATE bridge_dispatch_log
              SET outcome_score = $1, outcome_note = $2
              WHERE id = $3`,
             [score, note ?? null, id]
           );

           // Trigger async confidence refresh
           refreshConfidence().catch(() => {});

           return reply.send({ ok: true, dispatch_id: id, score, note: note ?? null });
         });
       }
       ```
