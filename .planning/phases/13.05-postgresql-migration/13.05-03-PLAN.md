<!-- FTS5 → tsvector conversion (critical for PG-03): -->
```typescript
// memory.ts FTS5 MATCH → tsvector @@
// BEFORE (SQLite FTS5):
sqlite.prepare(`
  SELECT c.* FROM concepts c
  JOIN concepts_fts f ON f.rowid = c.rowid
  WHERE concepts_fts MATCH ?
  ORDER BY rank
`).all(q)

// AFTER (PostgreSQL tsvector):
const { rows } = await pool.query(`
  SELECT *, ts_rank(search_vector, websearch_to_tsquery('english', $1)) AS rank
  FROM concepts
  WHERE search_vector @@ websearch_to_tsquery('english', $1)
    AND status = $2
  ORDER BY rank DESC
  LIMIT $3 OFFSET $4
`, [q, status, limit, offset]);

// conversations.ts messages FTS search:
// BEFORE:
sqlite.prepare(`
  SELECT m.* FROM messages m
  JOIN messages_fts f ON f.rowid = m.id
  WHERE messages_fts MATCH ?
  ORDER BY rank
`).all(q)

// AFTER:
const { rows } = await pool.query(`
  SELECT m.id, m.conversation_id, m.content, m.sender_name, m.sender_type,
         m.channel_type, m.created_at,
         ts_rank(m.search_vector, websearch_to_tsquery('english', $1)) AS rank
  FROM messages m
  WHERE m.search_vector @@ websearch_to_tsquery('english', $1)
  ORDER BY rank DESC
  LIMIT $2 OFFSET $3
`, [q, limit, offset]);
```

IMPORTANT: Use `websearch_to_tsquery` (NOT `to_tsquery`) for all user-facing search.
`to_tsquery` throws syntax errors on multi-word queries like "project manager".
`websearch_to_tsquery` safely handles spaces, phrases, and special characters.

<!-- Standard conversion patterns (same as Plan 02): -->
```typescript
// IMPORT: sqlite → pool
// QUERY: sqlite.prepare(sql).get/all/run() → await pool.query(sql, params)
// PARAMS: ? → $1, $2, $3...
// JSON: json_extract(col, '$.key') → col->>'key'
// TIME: unixepoch('now') → EXTRACT(EPOCH FROM NOW())
// UPSERT: INSERT OR IGNORE → INSERT ... ON CONFLICT DO NOTHING
```
