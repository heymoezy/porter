<!-- Conversion patterns (apply everywhere): -->
```typescript
// IMPORT: Change all files from:
import { sqlite } from '../db/client.js';  // or '../../db/client.js'
// To:
import { pool } from '../db/client.js';    // or '../../db/client.js'

// QUERY PATTERN: Change from:
const row = sqlite.prepare('SELECT * FROM x WHERE id = ?').get(id);
// To:
const { rows } = await pool.query('SELECT * FROM x WHERE id = $1', [id]);
const row = rows[0];

// ALL ROWS: Change from:
const rows = sqlite.prepare('SELECT * FROM x').all();
// To:
const { rows } = await pool.query('SELECT * FROM x');

// RUN (INSERT/UPDATE/DELETE): Change from:
sqlite.prepare('INSERT INTO x (a, b) VALUES (?, ?)').run(a, b);
// To:
await pool.query('INSERT INTO x (a, b) VALUES ($1, $2)', [a, b]);

// TRANSACTION: Change from:
const tx = sqlite.transaction(() => {
  sqlite.prepare(...).run(...);
  sqlite.prepare(...).run(...);
});
tx();
// To:
const client = await pool.connect();
try {
  await client.query('BEGIN');
  await client.query('INSERT INTO ...', [...]);
  await client.query('INSERT INTO ...', [...]);
  await client.query('COMMIT');
} catch (e) {
  await client.query('ROLLBACK');
  throw e;
} finally {
  client.release();
}

// JSON_EXTRACT: Change from:
json_extract(trigger_data, '$.contact_id')
// To:
trigger_data->>'contact_id'

// UNIXEPOCH: Change from:
unixepoch('now')
// To:
EXTRACT(EPOCH FROM NOW())

// NAMED PARAMS: Change from:
sqlite.prepare('... WHERE id = @workerId').get({ workerId: WORKER_ID })
// To:
await pool.query('... WHERE id = $1', [WORKER_ID])

// INSERT OR IGNORE: Change from:
INSERT OR IGNORE INTO ...
// To:
INSERT INTO ... ON CONFLICT DO NOTHING
```
