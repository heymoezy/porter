   **Column type conversion rules (apply to EVERY table):**
   | SQLite column | PostgreSQL column |
   |---|---|
   | `sqliteTable(...)` | `pgTable(...)` |
   | `real('x').default(sql\`(strftime('%s','now'))\`)` | `doublePrecision('x').default(sql\`EXTRACT(EPOCH FROM NOW())\`)` |
   | `real('x').default(sql\`(unixepoch('now'))\`)` | `doublePrecision('x').default(sql\`EXTRACT(EPOCH FROM NOW())\`)` |
   | `real('x')` (no default, used for timestamps) | `doublePrecision('x')` |
   | `integer('x').primaryKey({ autoIncrement: true })` | `serial('x').primaryKey()` |
   | `integer('x').default(0)` (boolean semantics: is_system, is_locked, etc.) | `integer('x').default(0)` (keep as integer per research recommendation) |
   | `integer('x').default(0)` (numeric: tokens, counts) | `integer('x').default(0)` (unchanged) |
   | `text('x').default('{}')` (JSON columns: metadata, config, trigger_data, etc.) | `jsonb('x').default(sql\`'{}'::jsonb\`)` |
   | `text('x').default('[]')` (JSON array columns: milestones, tags, skills, etc.) | `jsonb('x').default(sql\`'[]'::jsonb\`)` |
   | `blob('data')` | `bytea('data')` (custom import needed if not available — use `customType` or `text`) |
   | `text('x')` (plain text) | `text('x')` (unchanged) |

4. **Create `backend/src/db/migrate-consolidated.ts`:**
   This is the single-transaction DDL migration that creates all tables from scratch in PostgreSQL.

   Structure:
   ```typescript
   import { Pool } from 'pg';

   export async function migrateConsolidated(pool: Pool): Promise<void> {
     const client = await pool.connect();
     try {
       await client.query('BEGIN');

       // 1. Extensions
       await client.query('CREATE EXTENSION IF NOT EXISTS vector');

       // 2. schema_migrations table (first, for idempotency check)
       await client.query(`CREATE TABLE IF NOT EXISTS schema_migrations (
         id TEXT PRIMARY KEY,
         applied_at DOUBLE PRECISION DEFAULT EXTRACT(EPOCH FROM NOW())
       )`);

       // Idempotency: skip if already applied
       const check = await client.query(
         "SELECT 1 FROM schema_migrations WHERE id = 'consolidated_pg_v1'"
       );
       if (check.rows.length > 0) {
         await client.query('COMMIT');
         client.release();
         return;
       }

       // 3. All CREATE TABLE statements in FK dependency order:
       // users, sessions, tasks, chats, chat_messages, chat_attachments,
       // projects, personas, agent_jobs, agent_activity, decision_log,
       // token_usage_daily, workspace_connections, project_connections,
       // calendar_events, subscriptions, auth_tokens, billing_events,
       // project_collaborators, collaboration_events, companies, contacts,
       // contact_emails, contact_phones, contact_social, conversations,
       // messages (with search_vector tsvector), files, file_projects,
       // file_contacts, file_conversations, contact_conversations,
       // contact_projects, contact_analyses, agent_templates, concepts
       // (with search_vector tsvector), learning_sessions, errors

       // EACH TABLE must match the pgTable schema definitions exactly
       // Use IF NOT EXISTS for safety
       // All timestamp columns: DOUBLE PRECISION DEFAULT EXTRACT(EPOCH FROM NOW())
       // All JSON columns: JSONB DEFAULT '{}'::jsonb or '[]'::jsonb
       // All serial PKs: SERIAL PRIMARY KEY (or INTEGER GENERATED ALWAYS AS IDENTITY)
       // Boolean-semantic integers: INTEGER DEFAULT 0 (keep as integer)

       // 4. tsvector trigger functions
       await client.query(`
         CREATE OR REPLACE FUNCTION messages_search_update() RETURNS trigger AS $$
         BEGIN
           NEW.search_vector :=
             setweight(to_tsvector('english', COALESCE(NEW.content, '')), 'A') ||
             setweight(to_tsvector('english', COALESCE(NEW.sender_name, '')), 'B') ||
             setweight(to_tsvector('english', COALESCE(NEW.channel_type, '')), 'C');
           RETURN NEW;
         END;
         $$ LANGUAGE plpgsql
       `);
       await client.query(`
         CREATE TRIGGER messages_search_trig
           BEFORE INSERT OR UPDATE ON messages
           FOR EACH ROW EXECUTE FUNCTION messages_search_update()
       `);

       await client.query(`
         CREATE OR REPLACE FUNCTION concepts_search_update() RETURNS trigger AS $$
         BEGIN
           NEW.search_vector := to_tsvector('english', COALESCE(NEW.content, ''));
           RETURN NEW;
         END;
         $$ LANGUAGE plpgsql
       `);
       await client.query(`
         CREATE TRIGGER concepts_search_trig
           BEFORE INSERT OR UPDATE ON concepts
           FOR EACH ROW EXECUTE FUNCTION concepts_search_update()
       `);

       // 5. GIN indexes for FTS
       await client.query('CREATE INDEX IF NOT EXISTS idx_messages_search ON messages USING GIN(search_vector)');
       await client.query('CREATE INDEX IF NOT EXISTS idx_concepts_search ON concepts USING GIN(search_vector)');

       // 6. Other useful indexes (matching existing SQLite indexes)
       await client.query('CREATE INDEX IF NOT EXISTS idx_msg_conv ON messages(conversation_id)');
       await client.query('CREATE INDEX IF NOT EXISTS idx_msg_parent ON messages(parent_message_id) WHERE parent_message_id IS NOT NULL');
       await client.query('CREATE INDEX IF NOT EXISTS idx_jobs_status ON agent_jobs(status, scheduled_for)');
       await client.query('CREATE INDEX IF NOT EXISTS idx_activity_agent ON agent_activity(agent_id, created_at)');

       // 7. Mark migration complete
       await client.query(
         "INSERT INTO schema_migrations (id) VALUES ('consolidated_pg_v1') ON CONFLICT DO NOTHING"
       );

       await client.query('COMMIT');
     } catch (e) {
       await client.query('ROLLBACK');
       throw e;
     } finally {
       client.release();
     }
   }
   ```

   **CRITICAL:** The executor MUST write the full DDL for ALL tables from the schema.ts file. Every column, every constraint, every default. Do NOT leave placeholder comments — write the actual SQL.
