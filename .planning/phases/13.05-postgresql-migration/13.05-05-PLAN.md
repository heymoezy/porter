   b. If 0 rows, skip with log
   c. Transform types per row:
      - JSON TEXT columns (metadata, config, trigger_data, tags, skills, tools, etc.) -> `JSON.parse(value)` before INSERT (pg driver serializes objects to jsonb)

      - All other columns: pass through as-is

      ```typescript
      // For efficiency, build multi-row INSERT
      const cols = Object.keys(rows[0]);
      const placeholders = rows.map((_, i) =>
        `(${cols.map((_, j) => `$${i * cols.length + j + 1}`).join(', ')})`
      ).join(', ');
      const values = rows.flatMap(row => cols.map(c => row[c]));
      await pool.query(
        `INSERT INTO ${tableName} (${cols.join(', ')}) VALUES ${placeholders} ON CONFLICT DO NOTHING`,
        values
      );
      ```
      For large tables (1000+ rows), batch in chunks of 500 to avoid parameter limit.

   e. Log: `console.log(`Migrated ${tableName}: ${rows.length} rows`);`

      async function spotCheck(tableName: string, sqliteDb: Database, pool: Pool): Promise<{ok: number, fail: number}> {
        const sqliteRows = sqliteDb.prepare(`SELECT * FROM ${tableName} ORDER BY RANDOM() LIMIT 5`).all();
        let ok = 0, fail = 0;
        for (const row of sqliteRows) {
          // Find the primary key column (first column or 'id')
          const pkCol = Object.keys(row)[0];
          const pgResult = await pool.query(`SELECT * FROM ${tableName} WHERE ${pkCol} = $1`, [row[pkCol]]);
          if (pgResult.rows.length === 0) {
            console.error(`  MISSING: ${tableName} pk=${row[pkCol]}`);
            fail++;
          } else {
            // Compare each field
            const pgRow = pgResult.rows[0];
            let fieldMatch = true;
            for (const key of Object.keys(row)) {
              if (key === 'search_vector') continue; // skip tsvector columns
              const sqlVal = JSON.stringify(row[key]);
              const pgVal = JSON.stringify(pgRow[key]);
              if (sqlVal !== pgVal) {
                console.error(`  MISMATCH: ${tableName}.${key} pk=${row[pkCol]}: sqlite=${sqlVal} pg=${pgVal}`);
                fieldMatch = false;
              }
            }
            if (fieldMatch) ok++; else fail++;
          }
        }
        return { ok, fail };
      }
      ```
      Print spot-check summary per table. Fail the script if any table has spot-check failures.

      async function verifyFTS(pool: Pool): Promise<boolean> {
        // Get some actual content to search for
        const conceptSamples = await pool.query('SELECT content FROM concepts LIMIT 10');
        const messageSamples = await pool.query('SELECT content FROM messages WHERE content IS NOT NULL LIMIT 10');

        let allPassed = true;

        // Extract single words from actual content to use as search terms
        for (const row of conceptSamples.rows.slice(0, 5)) {
          const word = row.content?.split(/\s+/).find((w: string) => w.length > 3);
          if (!word) continue;
          const result = await pool.query(
            `SELECT COUNT(*) as cnt FROM concepts WHERE search_vector @@ websearch_to_tsquery('english', $1)`,
            [word]
          );
          const cnt = parseInt(result.rows[0].cnt);
          console.log(`  FTS concepts "${word}": ${cnt} results ${cnt > 0 ? 'OK' : 'EMPTY'}`);
          if (cnt === 0) allPassed = false;
        }

        for (const row of messageSamples.rows.slice(0, 5)) {
          const word = row.content?.split(/\s+/).find((w: string) => w.length > 3);
          if (!word) continue;
          const result = await pool.query(
            `SELECT COUNT(*) as cnt FROM messages WHERE search_vector @@ websearch_to_tsquery('english', $1)`,
            [word]
          );
          const cnt = parseInt(result.rows[0].cnt);
          console.log(`  FTS messages "${word}": ${cnt} results ${cnt > 0 ? 'OK' : 'EMPTY'}`);
          if (cnt === 0) allPassed = false;
        }

        return allPassed;
      }
      ```
