**Step 4 — Refactor `admin/frontend/app/hooks/use-admin-sse.ts` to use `addEventListener`:**

**CRITICAL background (pre-existing dormant bug):** Backend `sse-hub.ts:22` writes `event: <topic>\ndata: <json>` (named SSE events). The current hook uses `es.onmessage`, which ONLY fires for UNNAMED events. As a result, every existing case under the switch (bridge:dispatch, bridge:circuit-trip, intelligence:*, etc.) NEVER FIRES against the live backend. This plan fixes the dormant bug by switching to `addEventListener(topic, handler)` for each named topic — and the 3 new dreams topics inherit the correct behaviour for free.

Rewrite the hook so each topic gets its own listener. Replace the entire body of the `useEffect` (preserving the imports + the closure over `qc`) with:

```typescript
useEffect(() => {
  const es = new EventSource("/api/events")
  const adminEs = new EventSource("/api/admin/events")

  // Helper: register a handler that parses event.data once.
  const on = (src: EventSource, topic: string, handler: (payload: any) => void) => {
    const listener = (ev: MessageEvent) => {
      try {
        const payload = JSON.parse(ev.data)
        handler(payload)
      } catch {
        // Heartbeats or non-JSON — ignore
      }
    }
    src.addEventListener(topic, listener as EventListener)
  }

  // ─── /api/events channel ────────────────────────────────────────────────
  on(es, "profile:updated", (payload) => {
    const username = payload.data?.username ?? payload.username
    if (username) qc.invalidateQueries({ queryKey: ["admin", "customers", username] })
    qc.invalidateQueries({ queryKey: ["admin", "customers"] })
  })

  on(es, "agent:activity", () => qc.invalidateQueries({ queryKey: ["admin", "customers"] }))
  on(es, "agent:status",   () => qc.invalidateQueries({ queryKey: ["admin", "customers"] }))

  on(es, "bridge:health", () => {
    qc.invalidateQueries({ queryKey: ["bridge", "gateway-cards"] })
    qc.invalidateQueries({ queryKey: ["bridge", "capacity"] })
  })

  on(es, "bridge:usage", () => qc.invalidateQueries({ queryKey: ["bridge", "capacity"] }))

  on(es, "bridge:activity", (payload) => {
    qc.invalidateQueries({ queryKey: ["bridge", "capacity"] })
    window.dispatchEvent(new CustomEvent("bridge:activity", { detail: payload.data ?? payload }))
  })

  on(es, "bridge:dispatch", (payload) => {
    const entry = payload.data?.entry ?? payload.entry
    if (entry) {
      const key = ["bridge", "dispatch-log", 1, 50, "", "", ""]
      const existing = qc.getQueryData<DispatchLogCache>(key)
      if (existing) {
        qc.setQueryData<DispatchLogCache>(key, {
          ...existing,
          entries: [entry, ...existing.entries].slice(0, 50),
          pagination: {
            ...existing.pagination,
            total: existing.pagination.total + 1,
          },
        })
      } else {
        qc.invalidateQueries({ queryKey: ["bridge", "dispatch-log"] })
      }
    }
    qc.invalidateQueries({ queryKey: ["bridge", "dispatch-log-summary"] })
    qc.invalidateQueries({ queryKey: ["bridge", "costs-summary"] })
    qc.invalidateQueries({ queryKey: ["bridge", "intel-recent"] })
    qc.invalidateQueries({ queryKey: ["bridge", "capacity"] })
  })

  on(es, "bridge:circuit-trip", (payload) => {
    qc.invalidateQueries({ queryKey: ["bridge", "gateway-cards"] })
    window.dispatchEvent(new CustomEvent("bridge:circuit-trip", { detail: payload.data ?? payload }))
  })

  on(es, "bridge:context-pressure", (payload) => {
    qc.invalidateQueries({ queryKey: ["bridge", "sessions"] })
    window.dispatchEvent(new CustomEvent("bridge:context-pressure", { detail: payload.data ?? payload }))
  })

  on(es, "bridge:intelligence", (payload) => {
    qc.invalidateQueries({ queryKey: ["bridge", "patterns"] })
    window.dispatchEvent(new CustomEvent("bridge:intelligence", { detail: payload.data ?? payload }))
  })

  on(es, "bridge:msg-bus", (payload) => {
    qc.invalidateQueries({ queryKey: ["bridge", "msgbus"] })
    window.dispatchEvent(new CustomEvent("bridge:msg-bus", { detail: payload.data ?? payload }))
  })

  // ─── Phase 48.4 — Dreams topics ─────────────────────────────────────────
  const onProposalsChange = () => {
    qc.invalidateQueries({ queryKey: ["admin", "dreams", "proposals"] })
    qc.invalidateQueries({ queryKey: ["admin", "dreams", "runs"] })
  }
  on(es, "proposals:created",  onProposalsChange)
  on(es, "proposals:resolved", onProposalsChange)
  on(es, "dreams:run-completed", (payload) => {
    qc.invalidateQueries({ queryKey: ["admin", "dreams"] })
    window.dispatchEvent(new CustomEvent("dreams:run-completed", { detail: payload.data ?? payload }))
  })

  // ─── /api/admin/events channel ──────────────────────────────────────────
  const onBridgeConfigChange = () => {
    qc.invalidateQueries({ queryKey: ["bridge"] })
    qc.invalidateQueries({ queryKey: ["admin", "intelligence"] })
  }
  on(adminEs, "bridge:config-changed", onBridgeConfigChange)
  on(adminEs, "bridge:restarted",      onBridgeConfigChange)
  on(adminEs, "bridge:updated",        onBridgeConfigChange)

  es.onerror = () => { /* EventSource auto-reconnects */ }
  adminEs.onerror = () => { /* EventSource auto-reconnects */ }

  return () => { es.close(); adminEs.close() }
}, [qc])
```

// ──────────────────────────────────────────────────────────────────────────
// fmtRel — relative-time helper (matches the approvals.tsx inline pattern)
// ──────────────────────────────────────────────────────────────────────────
function fmtRel(ts: number | null | undefined): string {
  if (!ts) return '—';
  const now = Date.now() / 1000;
  const diff = now - ts;
  if (diff < 60) return 'just now';
  if (diff < 3600) return `${Math.floor(diff / 60)}m ago`;
  if (diff < 86400) return `${Math.floor(diff / 3600)}h ago`;
  if (diff < 86400 * 7) return `${Math.floor(diff / 86400)}d ago`;
  return new Date(ts * 1000).toLocaleDateString();
}

function truncate(s: string, n = 120): string {
  if (!s) return '';
  return s.length > n ? s.slice(0, n - 1) + '…' : s;
}
